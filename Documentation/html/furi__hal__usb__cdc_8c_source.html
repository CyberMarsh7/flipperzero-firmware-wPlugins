431static void cdc_init(usbd_device* dev, FuriHalUsbInterface* intf, void* ctx) {
432 UNUSED(ctx);
433 usb_dev = dev;
434 cdc_if_cur = intf;
435
436 char* name = (char*)furi_hal_version_get_device_name_ptr();
437 uint8_t len = (name == NULL) ? (0) : (strlen(name));
438 struct usb_string_descriptor* dev_prod_desc = malloc(len * 2 + 2);
439 dev_prod_desc->bLength = len * 2 + 2;
440 dev_prod_desc->bDescriptorType = USB_DTYPE_STRING;
441 for(uint8_t i = 0; i < len; i++) dev_prod_desc->wString[i] = name[i];
442
443 name = (char*)furi_hal_version_get_name_ptr();
444 len = (name == NULL) ? (0) : (strlen(name));
445 struct usb_string_descriptor* dev_serial_desc = malloc((len + 5) * 2 + 2);
446 dev_serial_desc->bLength = (len + 5) * 2 + 2;
447 dev_serial_desc->bDescriptorType = USB_DTYPE_STRING;
448 memcpy(dev_serial_desc->wString, "f\0l\0i\0p\0_\0", 5 * 2);
449 for(uint8_t i = 0; i < len; i++) dev_serial_desc->wString[i + 5] = name[i];
450
451 cdc_if_cur->str_prod_descr = dev_prod_desc;
452 cdc_if_cur->str_serial_descr = dev_serial_desc;
453
454 usbd_reg_config(dev, cdc_ep_config);
455 usbd_reg_control(dev, cdc_control);
456
457 usbd_connect(dev, true);
458}

577/* Configure endpoints */
578static usbd_respond cdc_ep_config(usbd_device* dev, uint8_t cfg) {
579 uint8_t if_cnt = ((struct usb_config_descriptor*)(cdc_if_cur->cfg_descr))->bNumInterfaces;
580 switch(cfg) {
581 case 0:
582 /* deconfiguring device */
583 if(if_cnt == 4) {
584 usbd_ep_deconfig(dev, CDC1_NTF_EP);
585 usbd_ep_deconfig(dev, CDC1_TXD_EP);
586 usbd_ep_deconfig(dev, CDC1_RXD_EP);
587 usbd_reg_endpoint(dev, CDC1_RXD_EP, 0);
588 usbd_reg_endpoint(dev, CDC1_TXD_EP, 0);
589 }
590 usbd_ep_deconfig(dev, CDC0_NTF_EP);
591 usbd_ep_deconfig(dev, CDC0_TXD_EP);
592 usbd_ep_deconfig(dev, CDC0_RXD_EP);
593 usbd_reg_endpoint(dev, CDC0_RXD_EP, 0);
594 usbd_reg_endpoint(dev, CDC0_TXD_EP, 0);
595 return usbd_ack;
596 case 1:
597 /* configuring device */
598 if((CDC0_TXD_EP & 0x7F) != (CDC0_RXD_EP & 0x7F)) {
599 // 2x unidirectional endpoint mode with dualbuf
600 usbd_ep_config(dev, CDC0_RXD_EP, USB_EPTYPE_BULK | USB_EPTYPE_DBLBUF, CDC_DATA_SZ);
601 usbd_ep_config(dev, CDC0_TXD_EP, USB_EPTYPE_BULK | USB_EPTYPE_DBLBUF, CDC_DATA_SZ);
602 usbd_ep_config(dev, CDC0_NTF_EP, USB_EPTYPE_INTERRUPT, CDC_NTF_SZ);
603 usbd_reg_endpoint(dev, CDC0_RXD_EP, cdc_rx_ep_callback);
604 usbd_reg_endpoint(dev, CDC0_TXD_EP, cdc_tx_ep_callback);
605 } else {
606 // 1x bidirectional endpoint mode
607 usbd_ep_config(dev, CDC0_RXD_EP, USB_EPTYPE_BULK, CDC_DATA_SZ);
608 usbd_ep_config(dev, CDC0_TXD_EP, USB_EPTYPE_BULK, CDC_DATA_SZ);
609 usbd_ep_config(dev, CDC0_NTF_EP, USB_EPTYPE_INTERRUPT, CDC_NTF_SZ);
610 usbd_reg_endpoint(dev, CDC0_RXD_EP, cdc_txrx_ep_callback);
611 usbd_reg_endpoint(dev, CDC0_TXD_EP, cdc_txrx_ep_callback);
612 }
613 usbd_ep_write(dev, CDC0_TXD_EP, 0, 0);
614
615 if(if_cnt == 4) {

639 UNUSED(callback);
640 /* CDC control requests */
641 uint8_t if_num = 0;
642 if(((USB_REQ_RECIPIENT | USB_REQ_TYPE) & req->bmRequestType) ==
643 (USB_REQ_INTERFACE | USB_REQ_CLASS) &&
644 (req->wIndex == 0 || req->wIndex == 2)) {
645 if(req->wIndex == 0)
646 if_num = 0;
647 else
648 if_num = 1;
649
650 switch(req->bRequest) {
651 case USB_CDC_SET_CONTROL_LINE_STATE:
652 if(callbacks[if_num] != NULL) {
653 cdc_ctrl_line_state[if_num] = req->wValue;
654 if(callbacks[if_num]->ctrl_line_callback != NULL)
655 callbacks[if_num]->ctrl_line_callback(
656 cb_ctx[if_num], cdc_ctrl_line_state[if_num]);
657 }
658 return usbd_ack;
659 case USB_CDC_SET_LINE_CODING:
660 memcpy(&cdc_config[if_num], req->data, sizeof(cdc_config[0]));
661 if(callbacks[if_num] != NULL) {
662 if(callbacks[if_num]->config_callback != NULL)
663 callbacks[if_num]->config_callback(cb_ctx[if_num], &cdc_config[if_num]);
664 }
665 return usbd_ack;
666 case USB_CDC_GET_LINE_CODING:
667 dev->status.data_ptr = &cdc_config[if_num];
668 dev->status.data_count = sizeof(cdc_config[0]);
669 return usbd_ack;
670 default:
671 return usbd_fail;
672 }
673 }
674 return usbd_fail;
