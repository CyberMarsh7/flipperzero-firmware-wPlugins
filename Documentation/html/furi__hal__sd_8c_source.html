342static SdSpiCmdAnswer
343 sd_spi_send_cmd(SdSpiCmd cmd, uint32_t arg, uint8_t crc, SdSpiCmdAnswerType answer_type) {
344 uint8_t frame[SD_CMD_LENGTH];
345 SdSpiCmdAnswer cmd_answer = {
346 .r1 = SD_DUMMY_BYTE,
347 .r2 = SD_DUMMY_BYTE,
348 .r3 = SD_DUMMY_BYTE,
349 .r4 = SD_DUMMY_BYTE,
350 .r5 = SD_DUMMY_BYTE,
351 };
352
353 // R1 Length = NCS(0)+ 6 Bytes command + NCR(min1 max8) + 1 Bytes answer + NEC(0) = 15bytes
354 // R1b identical to R1 + Busy information
355 // R2 Length = NCS(0)+ 6 Bytes command + NCR(min1 max8) + 2 Bytes answer + NEC(0) = 16bytes
356
357 frame[0] = ((uint8_t)cmd | 0x40);
358 frame[1] = (uint8_t)(arg >> 24);
359 frame[2] = (uint8_t)(arg >> 16);
360 frame[3] = (uint8_t)(arg >> 8);
361 frame[4] = (uint8_t)(arg);
362 frame[5] = (crc | 0x01);
363
364 sd_spi_select_card();
365 sd_spi_write_bytes(frame, sizeof(frame));
366
367 switch(answer_type) {
368 case SdSpiCmdAnswerTypeR1:
369 cmd_answer.r1 = sd_spi_wait_for_data_and_read();
370 break;
371 case SdSpiCmdAnswerTypeR1B:
372 // TODO FL-3507: can be wrong, at least for SD_CMD12_STOP_TRANSMISSION you need to purge one byte before reading R1
373 cmd_answer.r1 = sd_spi_wait_for_data_and_read();
374
375 // In general this shenenigans seems suspicious, please double check SD specs if you are using SdSpiCmdAnswerTypeR1B
376 // reassert card
377 sd_spi_deselect_card();
378 furi_delay_us(1000);
379 sd_spi_deselect_card();
380
381 // and wait for it to be ready
382 while(sd_spi_read_byte() != 0xFF) {
383 };
384
385 break;
386 case SdSpiCmdAnswerTypeR2:
387 cmd_answer.r1 = sd_spi_wait_for_data_and_read();
388 cmd_answer.r2 = sd_spi_read_byte();
389 break;
390 case SdSpiCmdAnswerTypeR3:
391 case SdSpiCmdAnswerTypeR7:
392 cmd_answer.r1 = sd_spi_wait_for_data_and_read();
393 cmd_answer.r2 = sd_spi_read_byte();
394 cmd_answer.r3 = sd_spi_read_byte();
395 cmd_answer.r4 = sd_spi_read_byte();
396 cmd_answer.r5 = sd_spi_read_byte();
397 break;
398 default:
399 break;
400 }
401 return cmd_answer;
402}

509static FuriStatus sd_spi_init_spi_mode(void) {
510 SdSpiCmdAnswer response;
511 uint8_t retry_count;
512
513 // CMD0 (GO_IDLE_STATE) to put SD in SPI mode and
514 // wait for In Idle State Response (R1 Format) equal to 0x01
515 retry_count = 0;
516 do {
517 retry_count++;
518 response = sd_spi_send_cmd(SD_CMD0_GO_IDLE_STATE, 0, 0x95, SdSpiCmdAnswerTypeR1);
519 sd_spi_deselect_card_and_purge();
520
521 if(retry_count >= SD_IDLE_RETRY_COUNT) {
522 sd_spi_debug("CMD0 failed");
523 return FuriStatusError;
524 }
525 } while(response.r1 != SdSpi_R1_IN_IDLE_STATE);
526
527 // CMD8 (SEND_IF_COND) to check the power supply status
528 // and wait until response (R7 Format) equal to 0xAA and
529 response = sd_spi_send_cmd(SD_CMD8_SEND_IF_COND, 0x1AA, 0x87, SdSpiCmdAnswerTypeR7);
530 sd_spi_deselect_card_and_purge();
531
532 if(FLAG_SET(response.r1, SdSpi_R1_ILLEGAL_COMMAND)) {
533 if(sd_spi_init_spi_mode_v1() != FuriStatusOk) {
534 sd_spi_debug("Init mode v1 failed");
535 return FuriStatusError;
536 }
537 sd_high_capacity = 0;
538 } else if(response.r1 == SdSpi_R1_IN_IDLE_STATE) {
539 if(sd_spi_init_spi_mode_v2() != FuriStatusOk) {
540 sd_spi_debug("Init mode v2 failed");
541 return FuriStatusError;
542 }
543
544 // CMD58 (READ_OCR) to initialize SDHC or SDXC cards: R3 response
545 response = sd_spi_send_cmd(SD_CMD58_READ_OCR, 0, 0xFF, SdSpiCmdAnswerTypeR3);
546 sd_spi_deselect_card_and_purge();
547
548 if(response.r1 != SdSpi_R1_NO_ERROR) {
549 sd_spi_debug("CMD58 failed");
550 return FuriStatusError;
551 }
552 sd_high_capacity = (response.r2 & 0x40) >> 6;
553 } else {
554 return FuriStatusError;
555 }
556
557 sd_spi_debug("SD card is %s", sd_high_capacity ? "SDHC or SDXC" : "SDSC");
558 return FuriStatusOk;

561static FuriStatus sd_spi_get_csd(SD_CSD* csd) {
562 uint16_t counter = 0;
563 uint8_t csd_data[16];
564 FuriStatus ret = FuriStatusError;
565 SdSpiCmdAnswer response;
566
567 // CMD9 (SEND_CSD): R1 format (0x00 is no errors)
568 response = sd_spi_send_cmd(SD_CMD9_SEND_CSD, 0, 0xFF, SdSpiCmdAnswerTypeR1);
569
570 if(response.r1 == SdSpi_R1_NO_ERROR) {
571 if(sd_spi_wait_for_data(SD_TOKEN_START_DATA_SINGLE_BLOCK_READ, SD_TIMEOUT_MS) ==
572 FuriStatusOk) {
573 // read CSD data
574 for(counter = 0; counter < 16; counter++) {
575 csd_data[counter] = sd_spi_read_byte();
576 }
577
578 sd_spi_purge_crc();
579
580 /*************************************************************************
581 CSD header decoding
582 *************************************************************************/
583
584 csd->CSDStruct = (csd_data[0] & 0xC0) >> 6;
585 csd->Reserved1 = csd_data[0] & 0x3F;
586 csd->TAAC = csd_data[1];
587 csd->NSAC = csd_data[2];
588 csd->MaxBusClkFreq = csd_data[3];
589 csd->CardComdClasses = (csd_data[4] << 4) | ((csd_data[5] & 0xF0) >> 4);
590 csd->RdBlockLen = csd_data[5] & 0x0F;

688static FuriStatus
689 sd_spi_cmd_read_blocks(uint32_t* data, uint32_t address, uint32_t blocks, uint32_t timeout_ms) {
690 uint32_t block_address = address;
691 uint32_t offset = 0;
692
693 // CMD16 (SET_BLOCKLEN): R1 response (0x00: no errors)
694 SdSpiCmdAnswer response =
695 sd_spi_send_cmd(SD_CMD16_SET_BLOCKLEN, SD_BLOCK_SIZE, 0xFF, SdSpiCmdAnswerTypeR1);
696 sd_spi_deselect_card_and_purge();
697
698 if(response.r1 != SdSpi_R1_NO_ERROR) {
699 return FuriStatusError;
700 }
701
702 if(!sd_high_capacity) {
703 block_address = address * SD_BLOCK_SIZE;
704 }
705
706 while(blocks--) {
707 // CMD17 (READ_SINGLE_BLOCK): R1 response (0x00: no errors)
708 response =
709 sd_spi_send_cmd(SD_CMD17_READ_SINGLE_BLOCK, block_address, 0xFF, SdSpiCmdAnswerTypeR1);
710 if(response.r1 != SdSpi_R1_NO_ERROR) {
711 sd_spi_deselect_card_and_purge();
712 return FuriStatusError;
713 }
714
715 // Wait for the data start token
716 if(sd_spi_wait_for_data(SD_TOKEN_START_DATA_SINGLE_BLOCK_READ, timeout_ms) ==
717 FuriStatusOk) {
718 // Read the data block
719 sd_spi_read_bytes_dma((uint8_t*)data + offset, SD_BLOCK_SIZE);
720 sd_spi_purge_crc();
721
722 // increase offset
723 offset += SD_BLOCK_SIZE;
724
725 // increase block address
726 if(sd_high_capacity) {
727 block_address += 1;
728 } else {
729 block_address += SD_BLOCK_SIZE;
730 }
731 } else {
732 sd_spi_deselect_card_and_purge();
733 return FuriStatusError;
734 }
735
736 sd_spi_deselect_card_and_purge();
737 }
738
739 return FuriStatusOk;
740}
