129bool furi_hal_spi_bus_trx(
130 FuriHalSpiBusHandle* handle,
131 const uint8_t* tx_buffer,
132 uint8_t* rx_buffer,
133 size_t size,
134 uint32_t timeout) {
135 furi_assert(handle);
136 furi_assert(handle->bus->current_handle == handle);
137 furi_assert(size > 0);
138
139 bool ret = true;
140 size_t tx_size = size;
141 bool tx_allowed = true;
142
143 while(size > 0) {
144 if(tx_size > 0 && LL_SPI_IsActiveFlag_TXE(handle->bus->spi) && tx_allowed) {
145 if(tx_buffer) {
146 LL_SPI_TransmitData8(handle->bus->spi, *tx_buffer);
147 tx_buffer++;
148 } else {
149 LL_SPI_TransmitData8(handle->bus->spi, 0xFF);
150 }
151 tx_size--;
152 tx_allowed = false;
153 }
154
155 if(LL_SPI_IsActiveFlag_RXNE(handle->bus->spi)) {
156 if(rx_buffer) {
157 *rx_buffer = LL_SPI_ReceiveData8(handle->bus->spi);
158 rx_buffer++;
159 } else {
160 LL_SPI_ReceiveData8(handle->bus->spi);
161 }
162 size--;
163 tx_allowed = true;
164 }
165 }
166
167 furi_hal_spi_bus_end_txrx(handle, timeout);
168
169 return ret;
170}
