41uint16_t DRV_CANFDSPI_CalculateCRC16(uint8_t* data, uint16_t size) {
42 uint16_t init = CRCBASE;
43 uint8_t index;
44
45 while(size-- != 0) {
46 index = ((uint8_t*)&init)[CRCUPPER] ^ *data++;
47 init = (init << 8) ^ crc16_table[index];

403int8_t mcp2518fd::mcp2518fd_ReadByteArrayWithCRC(
404 uint16_t address,
405 uint8_t* rxd,
406 uint16_t nBytes,
407 bool fromRam,
408 bool* crcIsCorrect) {
409 uint8_t i;
410 uint16_t crcFromSpiSlave = 0;
411 uint16_t crcAtController = 0;
412 uint16_t spiTransferSize =
413 nBytes + 5; // first two bytes for sending command & address, third for
414 // size, last two bytes for CRC
415 int8_t spiTransferError = 0;
416
417 // Compose command
418 spiTransmitBuffer[0] = (uint8_t)((cINSTRUCTION_READ_CRC << 4) + ((address >> 8) & 0xF));
419 spiTransmitBuffer[1] = (uint8_t)(address & 0xFF);
420 spiTransmitBuffer[2] = fromRam ? (nBytes >> 2) : nBytes;
421
422 // Clear data
423 for(i = 3; i < spiTransferSize; i++) {
424 spiTransmitBuffer[i] = 0;
425 }
426
427#ifdef SPI_HAS_TRANSACTION
428 SPI_BEGIN();
429#endif
430 MCP2518fd_SELECT();
431 spi_readwrite(spiTransmitBuffer[0]);
432 spi_readwrite(spiTransmitBuffer[1]);
433 spi_readwrite(spiTransmitBuffer[2]);
434 for(i = 3; i < spiTransferSize; i++) {
435 spiReceiveBuffer[i] = spi_readwrite(0x00);
436 }
437 MCP2518fd_UNSELECT();
438#ifdef SPI_HAS_TRANSACTION
439 SPI_END();
440#endif
441
442 // Get CRC from controller
443 crcFromSpiSlave = (uint16_t)(spiReceiveBuffer[spiTransferSize - 2] << 8) +
444 (uint16_t)(spiReceiveBuffer[spiTransferSize - 1]);
445
446 // Use the receive buffer to calculate CRC
447 // First three bytes need to be command
448 spiReceiveBuffer[0] = spiTransmitBuffer[0];
449 spiReceiveBuffer[1] = spiTransmitBuffer[1];
450 spiReceiveBuffer[2] = spiTransmitBuffer[2];
451 crcAtController = DRV_CANFDSPI_CalculateCRC16(spiReceiveBuffer, nBytes + 3);
452
453 // Compare CRC readings
454 if(crcFromSpiSlave == crcAtController) {
455 *crcIsCorrect = true;
456 } else {
457 *crcIsCorrect = false;
458 }
459
460 // Update data
461 for(i = 0; i < nBytes; i++) {
462 rxd[i] = spiReceiveBuffer[i + 3];
463 }
464
465 return spiTransferError;
466}

801 * bittime calculation code from
802 * https://github.com/pierremolinaro/acan2517FD
803 *
804 */

806static const uint16_t MAX_BRP = 256;
807static const uint16_t MAX_ARBITRATION_PHASE_SEGMENT_1 = 256;
808static const uint8_t MAX_ARBITRATION_PHASE_SEGMENT_2 = 128;
809static const uint8_t MAX_ARBITRATION_SJW = 128;
810static const uint16_t MAX_DATA_PHASE_SEGMENT_1 = 32;
811static const uint8_t MAX_DATA_PHASE_SEGMENT_2 = 16;
812static const uint8_t MAX_DATA_SJW = 16;
813
814int mcp2518fd::calcBittime(
815 const uint32_t inDesiredArbitrationBitRate,
816 const uint32_t inTolerancePPM) {
817 if(mDataBitRateFactor <= 1) { // Single bit rate
818 const uint32_t maxTQCount = MAX_ARBITRATION_PHASE_SEGMENT_1 +
819 MAX_ARBITRATION_PHASE_SEGMENT_2 +
820 1; // Setting for slowest bit rate
821 uint32_t BRP = MAX_BRP;
822 uint32_t smallestError = UINT32_MAX;
823 uint32_t bestBRP = 1; // Setting for highest bit rate
824 uint32_t bestTQCount = 4; // Setting for highest bit rate
825 uint32_t TQCount = mSysClock / inDesiredArbitrationBitRate / BRP;
826 //--- Loop for finding best BRP and best TQCount
827 while(
828 (TQCount <= (MAX_ARBITRATION_PHASE_SEGMENT_1 + MAX_ARBITRATION_PHASE_SEGMENT_2 + 1)) &&
829 (BRP > 0)) {
830 //--- Compute error using TQCount
831 if((TQCount >= 4) && (TQCount <= maxTQCount)) {
832 const uint32_t error = mSysClock - inDesiredArbitrationBitRate * TQCount *
833 BRP; // error is always >= 0
834 if(error <= smallestError) {
835 smallestError = error;
836 bestBRP = BRP;
837 bestTQCount = TQCount;
838 }
839 }
840 //--- Compute error using TQCount+1
841 if((TQCount >= 3) && (TQCount < maxTQCount)) {
842 const uint32_t error = inDesiredArbitrationBitRate * (TQCount + 1) * BRP -
843 mSysClock; // error is always >= 0
844 if(error <= smallestError) {
845 smallestError = error;
846 bestBRP = BRP;
847 bestTQCount = TQCount + 1;
848 }
849 }
850 //--- Continue with next value of BRP
851 BRP--;
852 TQCount = (BRP == 0) ? (maxTQCount + 1) :
853 (mSysClock / inDesiredArbitrationBitRate / BRP);
854 }
855 //--- Compute PS2 (1 <= PS2 <= 128)
856 uint32_t PS2 = bestTQCount / 5; // For sampling point at 80%
857 if(PS2 == 0) {
858 PS2 = 1;
859 } else if(PS2 > MAX_ARBITRATION_PHASE_SEGMENT_2) {
860 PS2 = MAX_ARBITRATION_PHASE_SEGMENT_2;
861 }
862 //--- Compute PS1 (1 <= PS1 <= 256)
863 uint32_t PS1 = bestTQCount - PS2 - 1 /* Sync Seg */;
864 if(PS1 > MAX_ARBITRATION_PHASE_SEGMENT_1) {
865 PS2 += PS1 - MAX_ARBITRATION_PHASE_SEGMENT_1;
866 PS1 = MAX_ARBITRATION_PHASE_SEGMENT_1;
867 }
868 //---
869 mBitRatePrescaler = (uint16_t)bestBRP;
870 mArbitrationPhaseSegment1 = (uint16_t)PS1;
871 mArbitrationPhaseSegment2 = (uint8_t)PS2;
872 mArbitrationSJW =
873 mArbitrationPhaseSegment2; // Always 1 <= SJW <= 128, and SJW <= mArbitrationPhaseSegment2
874 //--- Final check of the nominal configuration
875 const uint32_t W = bestTQCount * inDesiredArbitrationBitRate * bestBRP;
876 const uint64_t diff = (mSysClock > W) ? (mSysClock - W) : (W - mSysClock);
877 const uint64_t ppm = (uint64_t)(1000UL * 1000UL); // UL suffix is required for Arduino Uno
878 mArbitrationBitRateClosedToDesiredRate = (diff * ppm) <= (((uint64_t)W) * inTolerancePPM);
879 } else { // Dual bit rate, first compute data bit rate
880 const uint32_t maxDataTQCount =
881 MAX_DATA_PHASE_SEGMENT_1 + MAX_DATA_PHASE_SEGMENT_2; // Setting for slowest bit rate
882 const uint32_t desiredDataBitRate =
883 inDesiredArbitrationBitRate * uint8_t(mDataBitRateFactor);
884 uint32_t smallestError = UINT32_MAX;
885 uint32_t bestBRP = MAX_BRP; // Setting for lowest bit rate
886 uint32_t bestDataTQCount = maxDataTQCount; // Setting for lowest bit rate
887 uint32_t dataTQCount = 4;
888 uint32_t brp = mSysClock / desiredDataBitRate / dataTQCount;
889 //--- Loop for finding best BRP and best TQCount
890 while((dataTQCount <= maxDataTQCount) && (brp > 0)) {
891 //--- Compute error using brp
892 if(brp <= MAX_BRP) {
893 const uint32_t error =
894 mSysClock - desiredDataBitRate * dataTQCount * brp; // error is always >= 0
895 if(error <= smallestError) {
896 smallestError = error;
897 bestBRP = brp;
898 bestDataTQCount = dataTQCount;
899 }
900 }
901 //--- Compute error using brp+1
902 if(brp < MAX_BRP) {
903 const uint32_t error = desiredDataBitRate * dataTQCount * (brp + 1) -
904 mSysClock; // error is always >= 0
905 if(error <= smallestError) {
906 smallestError = error;
907 bestBRP = brp + 1;
908 bestDataTQCount = dataTQCount;
909 }
910 }
911 //--- Continue with next value of BRP
912 dataTQCount += 1;
913 brp = mSysClock / desiredDataBitRate / dataTQCount;
914 }
915 //--- Compute data PS2 (1 <= PS2 <= 16)
916 uint32_t dataPS2 = bestDataTQCount / 5; // For sampling point at 80%
917 if(dataPS2 == 0) {
918 dataPS2 = 1;
919 }
920 //--- Compute data PS1 (1 <= PS1 <= 32)
921 uint32_t dataPS1 = bestDataTQCount - dataPS2 - 1 /* Sync Seg */;
922 if(dataPS1 > MAX_DATA_PHASE_SEGMENT_1) {
923 dataPS2 += dataPS1 - MAX_DATA_PHASE_SEGMENT_1;
924 dataPS1 = MAX_DATA_PHASE_SEGMENT_1;
925 }
926 //---
927 const int TDCO = bestBRP * dataPS1; // According to DS20005678D, ??3.4.8 Page 20
928 mTDCO = (TDCO > 63) ? 63 : (int8_t)TDCO;
929 mDataPhaseSegment1 = (uint8_t)dataPS1;
930 mDataPhaseSegment2 = (uint8_t)dataPS2;
931 mDataSJW = mDataPhaseSegment2;
932 const uint32_t arbitrationTQCount = bestDataTQCount * uint8_t(mDataBitRateFactor);
933 //--- Compute arbiration PS2 (1 <= PS2 <= 128)
934 uint32_t arbitrationPS2 = arbitrationTQCount / 5; // For sampling point at 80%
935 if(arbitrationPS2 == 0) {
936 arbitrationPS2 = 1;
937 }
938 //--- Compute PS1 (1 <= PS1 <= 256)
939 uint32_t arbitrationPS1 = arbitrationTQCount - arbitrationPS2 - 1 /* Sync Seg */;
940 if(arbitrationPS1 > MAX_ARBITRATION_PHASE_SEGMENT_1) {
941 arbitrationPS2 += arbitrationPS1 - MAX_ARBITRATION_PHASE_SEGMENT_1;
942 arbitrationPS1 = MAX_ARBITRATION_PHASE_SEGMENT_1;
943 }
944 //---
945 mBitRatePrescaler = (uint16_t)bestBRP;
946 mArbitrationPhaseSegment1 = (uint16_t)arbitrationPS1;
947 mArbitrationPhaseSegment2 = (uint8_t)arbitrationPS2;
948 mArbitrationSJW =
949 mArbitrationPhaseSegment2; // Always 1 <= SJW <= 128, and SJW <= mArbitrationPhaseSegment2
950 //--- Final check of the nominal configuration
951 const uint32_t W = arbitrationTQCount * inDesiredArbitrationBitRate * bestBRP;
952 const uint64_t diff = (mSysClock > W) ? (mSysClock - W) : (W - mSysClock);
953 const uint64_t ppm = (uint64_t)(1000UL * 1000UL); // UL suffix is required for Arduino Uno
954 mArbitrationBitRateClosedToDesiredRate = (diff * ppm) <= (((uint64_t)W) * inTolerancePPM);
955 }
956 return mArbitrationBitRateClosedToDesiredRate;
957}

1327int8_t mcp2518fd::mcp2518fd_TransmitChannelLoad(
1328 CAN_FIFO_CHANNEL channel,
1329 CAN_TX_MSGOBJ* txObj,
1330 uint8_t* txd,
1331 uint32_t txdNumBytes,
1332 bool flush) {
1333 uint16_t a;
1334 uint32_t fifoReg[3];
1335 uint32_t dataBytesInObject;
1336 REG_CiFIFOCON ciFifoCon;
1337 REG_CiFIFOUA ciFifoUa;
1338 int8_t spiTransferError = 0;
1339
1340 // Get FIFO registers
1341 a = cREGADDR_CiFIFOCON + (channel * CiFIFO_OFFSET);
1342
1343 spiTransferError = mcp2518fd_ReadWordArray(a, fifoReg, 3);
1344 if(spiTransferError) {
1345 return -1;
1346 }
1347
1348 // Check that it is a transmit buffer
1349 ciFifoCon.word = fifoReg[0];
1350 if(!ciFifoCon.txBF.TxEnable) {
1351 return -2;
1352 }
1353
1354 // Check that DLC is big enough for data
1355 dataBytesInObject = DRV_CANFDSPI_DlcToDataBytes((CAN_DLC)txObj->bF.ctrl.DLC);
1356 if(dataBytesInObject < txdNumBytes) {
1357 return -3;
1358 }
1359
1360 // Get address
1361 ciFifoUa.word = fifoReg[2];
1362#ifdef USERADDRESS_TIMES_FOUR
1363 a = 4 * ciFifoUa.bF.UserAddress;
1364#else
1365 a = ciFifoUa.bF.UserAddress;
1366#endif
1367 a += cRAMADDR_START;
1368 uint8_t txBuffer[MAX_MSG_SIZE];
1369
1370 txBuffer[0] = txObj->byte[0]; // not using 'for' to reduce no of instructions
1371 txBuffer[1] = txObj->byte[1];
1372 txBuffer[2] = txObj->byte[2];
1373 txBuffer[3] = txObj->byte[3];
1374
1375 txBuffer[4] = txObj->byte[4];
1376 txBuffer[5] = txObj->byte[5];
1377 txBuffer[6] = txObj->byte[6];
1378 txBuffer[7] = txObj->byte[7];
1379
1380 uint8_t i;
1381 for(i = 0; i < txdNumBytes; i++) {
1382 txBuffer[i + 8] = txd[i];
1383 }
1384
1385 // Make sure we write a multiple of 4 bytes to RAM
1386 uint16_t n = 0;
1387 uint8_t j = 0;
1388
1389 if(txdNumBytes % 4) {
1390 // Need to add bytes
1391 n = 4 - (txdNumBytes % 4);
1392 i = txdNumBytes + 8;
1393
1394 for(j = 0; j < n; j++) {
1395 txBuffer[i + 8 + j] = 0;
1396 }
1397 }
1398 spiTransferError = mcp2518fd_WriteByteArray(a, txBuffer, txdNumBytes + 8 + n);
1399 if(spiTransferError) {
1400 return -4;
1401 }
1402
1403 // Set UINC and TXREQ
1404 spiTransferError = mcp2518fd_TransmitChannelUpdate(channel, flush);
1405 if(spiTransferError) {
1406 return -5;
1407 }
1408 return spiTransferError;
1409}

1521 } else {

1523
1524 // Assign message data
1525 for(i = 0; i < nBytes; i++) {
1526 rxd[i] = ba[i + 8];
1527 }
1528 }

1531 spiTransferError = mcp2518fd_ReceiveChannelUpdate(channel);
