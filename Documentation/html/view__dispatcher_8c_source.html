76void view_dispatcher_run(ViewDispatcher* view_dispatcher) {
77 furi_assert(view_dispatcher);
78 furi_assert(view_dispatcher->queue);
79
80 uint32_t tick_period = view_dispatcher->tick_period == 0 ? FuriWaitForever :
81 view_dispatcher->tick_period;
82 ViewDispatcherMessage message;
83 while(1) {
84 if(furi_message_queue_get(view_dispatcher->queue, &message, tick_period) != FuriStatusOk) {
85 view_dispatcher_handle_tick_event(view_dispatcher);
86 continue;
87 }
88 if(message.type == ViewDispatcherMessageTypeStop) {
89 break;
90 } else if(message.type == ViewDispatcherMessageTypeInput) {
91 view_dispatcher_handle_input(view_dispatcher, &message.input);
92 } else if(message.type == ViewDispatcherMessageTypeCustomEvent) {
93 view_dispatcher_handle_custom_event(view_dispatcher, message.custom_event);
94 }
95 }
96
97 // Wait till all input events delivered
98 while(view_dispatcher->ongoing_input) {
99 furi_message_queue_get(view_dispatcher->queue, &message, FuriWaitForever);
100 if(message.type == ViewDispatcherMessageTypeInput) {
101 uint8_t key_bit = (1 << message.input.key);
102 if(message.input.type == InputTypePress) {
103 view_dispatcher->ongoing_input |= key_bit;
104 } else if(message.input.type == InputTypeRelease) {
105 view_dispatcher->ongoing_input &= ~key_bit;
106 }
107 }
108 }

236void view_dispatcher_handle_input(ViewDispatcher* view_dispatcher, InputEvent* event) {
237 // Check input complementarity
238 uint8_t key_bit = (1 << event->key);
239 if(event->type == InputTypePress) {
240 view_dispatcher->ongoing_input |= key_bit;
241 } else if(event->type == InputTypeRelease) {
242 view_dispatcher->ongoing_input &= ~key_bit;
243 } else if(!(view_dispatcher->ongoing_input & key_bit)) {
244 FURI_LOG_D(
245 TAG,
246 "non-complementary input, discarding key: %s, type: %s, sequence: %p",
247 input_get_key_name(event->key),
248 input_get_type_name(event->type),
249 (void*)event->sequence);
250 return;
251 }
252
253 // Set ongoing input view if this is event is first press event
254 if(!(view_dispatcher->ongoing_input & ~key_bit) && event->type == InputTypePress) {
255 view_dispatcher->ongoing_input_view = view_dispatcher->current_view;
256 }
257
258 // Deliver event
259 if(view_dispatcher->current_view &&
260 view_dispatcher->ongoing_input_view == view_dispatcher->current_view) {
261 // Dispatch input to current view
262 bool is_consumed = view_input(view_dispatcher->current_view, event);
263
264 // Navigate if input is not consumed
265 if(!is_consumed && (event->key == InputKeyBack) &&
266 (event->type == InputTypeShort || event->type == InputTypeLong)) {
267 // Navigate to previous
268 uint32_t view_id = view_previous(view_dispatcher->current_view);
269 if(view_id != VIEW_IGNORE) {
270 // Switch to returned view
271 view_dispatcher_switch_to_view(view_dispatcher, view_id);
272 } else if(view_dispatcher->navigation_event_callback) {
273 // Dispatch navigation event
274 if(!view_dispatcher->navigation_event_callback(view_dispatcher->event_context)) {
275 view_dispatcher_stop(view_dispatcher);
276 return;
277 }
278 }
279 }
280 } else if(view_dispatcher->ongoing_input_view && event->type == InputTypeRelease) {
281 FURI_LOG_D(
282 TAG,
283 "View changed while key press %p -> %p. Sending key: %s, type: %s, sequence: %p to previous view port",
284 view_dispatcher->ongoing_input_view,
285 view_dispatcher->current_view,
286 input_get_key_name(event->key),
287 input_get_type_name(event->type),
288 (void*)event->sequence);
289 view_input(view_dispatcher->ongoing_input_view, event);
290 }
291}

329void view_dispatcher_set_current_view(ViewDispatcher* view_dispatcher, View* view) {
330 furi_assert(view_dispatcher);
331 // Dispatch view exit event
332 if(view_dispatcher->current_view) {
333 view_exit(view_dispatcher->current_view);
334 }
335 // Set current view
336 view_dispatcher->current_view = view;
337 // Dispatch view enter event
338 if(view_dispatcher->current_view) {
339 ViewPortOrientation orientation =
340 view_dispatcher_view_port_orientation_table[view->orientation];
341 if(view_port_get_orientation(view_dispatcher->view_port) != orientation) {
342 view_port_set_orientation(view_dispatcher->view_port, orientation);
343 // we just rotated input keys, now it's time to sacrifice some input
344 view_dispatcher->ongoing_input = 0;
345 }
346 view_enter(view_dispatcher->current_view);
347 view_port_enabled_set(view_dispatcher->view_port, true);
348 view_port_update(view_dispatcher->view_port);
349 } else {
350 view_port_enabled_set(view_dispatcher->view_port, false);
351 if(view_dispatcher->queue) {
352 view_dispatcher_stop(view_dispatcher);
