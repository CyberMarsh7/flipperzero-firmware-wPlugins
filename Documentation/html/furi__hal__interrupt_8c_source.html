125void furi_hal_interrupt_set_isr_ex(
126 FuriHalInterruptId index,
127 uint16_t priority,
128 FuriHalInterruptISR isr,
129 void* context) {
130 furi_check(index < FuriHalInterruptIdMax);
131 furi_check(priority <= 15);
132
133 if(isr) {
134 // Pre ISR set
135 furi_check(furi_hal_interrupt_isr[index].isr == NULL);
136 } else {
137 // Pre ISR clear
138 furi_hal_interrupt_disable(index);
139 furi_hal_interrupt_clear_pending(index);
140 }
141
142 furi_hal_interrupt_isr[index].isr = isr;
143 furi_hal_interrupt_isr[index].context = context;
144 __DMB();
145
146 if(isr) {
147 // Post ISR set
148 furi_hal_interrupt_clear_pending(index);
149 furi_hal_interrupt_enable(index, priority);
150 } else {
151 // Post ISR clear
152 }
153}
