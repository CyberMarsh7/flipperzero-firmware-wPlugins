100static void furi_hal_infrared_tim_rx_isr() {
101 static uint32_t previous_captured_ch2 = 0;
102
103 /* Timeout */
104 if(LL_TIM_IsActiveFlag_CC3(INFRARED_RX_TIMER)) {
105 LL_TIM_ClearFlag_CC3(INFRARED_RX_TIMER);
106 furi_assert(furi_hal_infrared_state == InfraredStateAsyncRx);
107
108 /* Timers CNT register starts to counting from 0 to ARR, but it is
109 * reseted when Channel 1 catches interrupt. It is not reseted by
110 * channel 2, though, so we have to distract it's values (see TimerIRQSourceCCI1 ISR).
111 * This can cause false timeout: when time is over, but we started
112 * receiving new signal few microseconds ago, because CNT register
113 * is reseted once per period, not per sample. */
114 if(LL_GPIO_IsInputPinSet(gpio_infrared_rx.port, gpio_infrared_rx.pin) != 0) {
115 if(infrared_tim_rx.timeout_callback)
116 infrared_tim_rx.timeout_callback(infrared_tim_rx.timeout_context);
117 }
118 }
119
120 /* Rising Edge */
121 if(LL_TIM_IsActiveFlag_CC1(INFRARED_RX_TIMER)) {
122 LL_TIM_ClearFlag_CC1(INFRARED_RX_TIMER);
123 furi_assert(furi_hal_infrared_state == InfraredStateAsyncRx);
124
125 if(READ_BIT(INFRARED_RX_TIMER->CCMR1, TIM_CCMR1_CC1S)) {
126 /* Low pin level is a Mark state of INFRARED signal. Invert level for further processing. */
127 uint32_t duration = LL_TIM_IC_GetCaptureCH1(INFRARED_RX_TIMER) - previous_captured_ch2;
128 if(infrared_tim_rx.capture_callback)
129 infrared_tim_rx.capture_callback(infrared_tim_rx.capture_context, 1, duration);
130 } else {
131 furi_crash();
132 }
133 }
134
135 /* Falling Edge */
136 if(LL_TIM_IsActiveFlag_CC2(INFRARED_RX_TIMER)) {
137 LL_TIM_ClearFlag_CC2(INFRARED_RX_TIMER);
138 furi_assert(furi_hal_infrared_state == InfraredStateAsyncRx);
139
140 if(READ_BIT(INFRARED_RX_TIMER->CCMR1, TIM_CCMR1_CC2S)) {
141 /* High pin level is a Space state of INFRARED signal. Invert level for further processing. */
142 uint32_t duration = LL_TIM_IC_GetCaptureCH2(INFRARED_RX_TIMER);
143 previous_captured_ch2 = duration;
144 if(infrared_tim_rx.capture_callback)
145 infrared_tim_rx.capture_callback(infrared_tim_rx.capture_context, 0, duration);
146 } else {
147 furi_crash();
148 }
149 }

287static void furi_hal_infrared_tx_dma_isr() {
288#if INFRARED_DMA_CH2_CHANNEL == LL_DMA_CHANNEL_2
289 if(LL_DMA_IsActiveFlag_TE2(INFRARED_DMA)) {
290 LL_DMA_ClearFlag_TE2(INFRARED_DMA);
291 furi_crash();
292 }
293 if(LL_DMA_IsActiveFlag_HT2(INFRARED_DMA) && LL_DMA_IsEnabledIT_HT(INFRARED_DMA_CH2_DEF)) {
294 LL_DMA_ClearFlag_HT2(INFRARED_DMA);
295 uint8_t buf_num = furi_hal_infrared_get_current_dma_tx_buffer();
296 uint8_t next_buf_num = !buf_num;
297 if(infrared_tim_tx.buffer[buf_num].last_packet_end) {
298 LL_DMA_DisableIT_HT(INFRARED_DMA_CH2_DEF);
299 } else if(
300 !infrared_tim_tx.buffer[buf_num].packet_end ||
301 (furi_hal_infrared_state == InfraredStateAsyncTx)) {
302 furi_hal_infrared_tx_fill_buffer(next_buf_num, 0);
303 if(infrared_tim_tx.buffer[next_buf_num].last_packet_end) {
304 LL_DMA_DisableIT_HT(INFRARED_DMA_CH2_DEF);
305 }
306 } else if(furi_hal_infrared_state == InfraredStateAsyncTxStopReq) {
307 /* fallthrough */
308 } else {
309 furi_crash();
310 }
311 }
312 if(LL_DMA_IsActiveFlag_TC2(INFRARED_DMA) && LL_DMA_IsEnabledIT_TC(INFRARED_DMA_CH2_DEF)) {
313 LL_DMA_ClearFlag_TC2(INFRARED_DMA);
314 furi_check(
315 (furi_hal_infrared_state == InfraredStateAsyncTxStopInProgress) ||
316 (furi_hal_infrared_state == InfraredStateAsyncTxStopReq) ||
317 (furi_hal_infrared_state == InfraredStateAsyncTx));
318
319 uint8_t buf_num = furi_hal_infrared_get_current_dma_tx_buffer();
320 uint8_t next_buf_num = !buf_num;
321 if(furi_hal_infrared_state == InfraredStateAsyncTxStopInProgress) {
322 furi_hal_infrared_tx_dma_terminate();
323 } else if(
324 infrared_tim_tx.buffer[buf_num].last_packet_end ||
325 (infrared_tim_tx.buffer[buf_num].packet_end &&
326 (furi_hal_infrared_state == InfraredStateAsyncTxStopReq))) {
327 furi_hal_infrared_state = InfraredStateAsyncTxStopInProgress;
328 furi_hal_infrared_tx_fill_buffer_last(next_buf_num);
329 furi_hal_infrared_tx_dma_set_buffer(next_buf_num);
330 } else {
331 /* if it's not end of the packet - continue receiving */
332 furi_hal_infrared_tx_dma_set_buffer(next_buf_num);
333 }
334 if(infrared_tim_tx.signal_sent_callback && infrared_tim_tx.buffer[buf_num].packet_end &&
335 (furi_hal_infrared_state != InfraredStateAsyncTxStopped)) {
336 infrared_tim_tx.signal_sent_callback(infrared_tim_tx.signal_sent_context);
337 }
338 }
339#else
340#error Update this code. Would you kindly?
341#endif
342}

447static void furi_hal_infrared_tx_fill_buffer_last(uint8_t buf_num) {
448 furi_assert(buf_num < 2);
449 furi_assert(furi_hal_infrared_state != InfraredStateAsyncRx);
450 furi_assert(furi_hal_infrared_state < InfraredStateMAX);
451 furi_assert(infrared_tim_tx.data_callback);
452 InfraredTxBuf* buffer = &infrared_tim_tx.buffer[buf_num];
453 furi_assert(buffer->data != NULL);
454 (void)buffer->data;
455 furi_assert(buffer->polarity != NULL);
456 (void)buffer->polarity;
457
458 infrared_tim_tx.buffer[buf_num].data[0] = 0; // 1 pulse
459 infrared_tim_tx.buffer[buf_num].polarity[0] = INFRARED_TX_CCMR_LOW;
460 infrared_tim_tx.buffer[buf_num].data[1] = 0; // 1 pulse
461 infrared_tim_tx.buffer[buf_num].polarity[1] = INFRARED_TX_CCMR_LOW;
462 infrared_tim_tx.buffer[buf_num].size = 2;
463 infrared_tim_tx.buffer[buf_num].last_packet_end = true;
464 infrared_tim_tx.buffer[buf_num].packet_end = true;

467static void furi_hal_infrared_tx_fill_buffer(uint8_t buf_num, uint8_t polarity_shift) {
468 furi_assert(buf_num < 2);
469 furi_assert(furi_hal_infrared_state != InfraredStateAsyncRx);
470 furi_assert(furi_hal_infrared_state < InfraredStateMAX);
471 furi_assert(infrared_tim_tx.data_callback);
472 InfraredTxBuf* buffer = &infrared_tim_tx.buffer[buf_num];
473 furi_assert(buffer->data != NULL);
474 furi_assert(buffer->polarity != NULL);
475
476 FuriHalInfraredTxGetDataState status = FuriHalInfraredTxGetDataStateOk;
477 uint32_t duration = 0;
478 bool level = 0;
479 size_t* size = &buffer->size;
480 size_t polarity_counter = 0;
481 while(polarity_shift--) {
482 buffer->polarity[polarity_counter++] = INFRARED_TX_CCMR_LOW;
483 }
484
485 for(*size = 0; (*size < INFRARED_TIM_TX_DMA_BUFFER_SIZE) &&
486 (status == FuriHalInfraredTxGetDataStateOk);) {
487 if(infrared_tim_tx.tx_timing_rest_duration > 0) {
488 if(infrared_tim_tx.tx_timing_rest_duration > 0xFFFF) {
489 buffer->data[*size] = 0xFFFF;
490 status = FuriHalInfraredTxGetDataStateOk;
491 } else {
492 buffer->data[*size] = infrared_tim_tx.tx_timing_rest_duration;
493 status = infrared_tim_tx.tx_timing_rest_status;
494 }
495 infrared_tim_tx.tx_timing_rest_duration -= buffer->data[*size];
496 buffer->polarity[polarity_counter] = infrared_tim_tx.tx_timing_rest_level ?
497 INFRARED_TX_CCMR_HIGH :
498 INFRARED_TX_CCMR_LOW;
499 ++(*size);
500 ++polarity_counter;
501 continue;
502 }
503
504 status = infrared_tim_tx.data_callback(infrared_tim_tx.data_context, &duration, &level);
505
506 uint32_t num_of_impulses = roundf(duration / infrared_tim_tx.cycle_duration);
507
508 if(num_of_impulses == 0) {
509 if((*size == 0) && (status == FuriHalInfraredTxGetDataStateDone)) {
510 /* if this is one sample in current buffer, but we
511 * have more to send - continue
512 */
513 status = FuriHalInfraredTxGetDataStateOk;
514 }
515 } else if((num_of_impulses - 1) > 0xFFFF) {
516 infrared_tim_tx.tx_timing_rest_duration = num_of_impulses - 1;
517 infrared_tim_tx.tx_timing_rest_status = status;
518 infrared_tim_tx.tx_timing_rest_level = level;
519 status = FuriHalInfraredTxGetDataStateOk;
520 } else {
521 buffer->polarity[polarity_counter] = level ? INFRARED_TX_CCMR_HIGH :
522 INFRARED_TX_CCMR_LOW;
523 buffer->data[*size] = num_of_impulses - 1;
524 ++(*size);
525 ++polarity_counter;
526 }
527 }
528
529 buffer->last_packet_end = (status == FuriHalInfraredTxGetDataStateLastDone);
530 buffer->packet_end = buffer->last_packet_end || (status == FuriHalInfraredTxGetDataStateDone);
531
532 if(*size == 0) {
533 buffer->data[0] = 0; // 1 pulse
534 buffer->polarity[0] = INFRARED_TX_CCMR_LOW;
535 buffer->size = 1;
536 }
537}

605void furi_hal_infrared_async_tx_start(uint32_t freq, float duty_cycle) {
606 if((duty_cycle > 1) || (duty_cycle <= 0) || (freq > INFRARED_MAX_FREQUENCY) ||
607 (freq < INFRARED_MIN_FREQUENCY) || (infrared_tim_tx.data_callback == NULL)) {
608 furi_crash();
609 }
610
611 furi_assert(furi_hal_infrared_state == InfraredStateIdle);
612 furi_assert(infrared_tim_tx.buffer[0].data == NULL);
613 furi_assert(infrared_tim_tx.buffer[1].data == NULL);
614 furi_assert(infrared_tim_tx.buffer[0].polarity == NULL);
615 furi_assert(infrared_tim_tx.buffer[1].polarity == NULL);
616
617 size_t alloc_size_data = INFRARED_TIM_TX_DMA_BUFFER_SIZE * sizeof(uint16_t);
618 infrared_tim_tx.buffer[0].data = malloc(alloc_size_data);
619 infrared_tim_tx.buffer[1].data = malloc(alloc_size_data);
620
621 size_t alloc_size_polarity =
622 (INFRARED_TIM_TX_DMA_BUFFER_SIZE + INFRARED_POLARITY_SHIFT) * sizeof(uint8_t);
623 infrared_tim_tx.buffer[0].polarity = malloc(alloc_size_polarity);
624 infrared_tim_tx.buffer[1].polarity = malloc(alloc_size_polarity);
625
626 infrared_tim_tx.stop_semaphore = furi_semaphore_alloc(1, 0);
627 infrared_tim_tx.cycle_duration = 1000000.0 / freq;
628 infrared_tim_tx.tx_timing_rest_duration = 0;
629
630 furi_hal_infrared_tx_fill_buffer(0, INFRARED_POLARITY_SHIFT);
631
632 furi_hal_bus_enable(INFRARED_DMA_TIMER_BUS);
633
634 furi_hal_infrared_configure_tim_pwm_tx(freq, duty_cycle);
635 furi_hal_infrared_configure_tim_cmgr2_dma_tx();
636 furi_hal_infrared_configure_tim_rcr_dma_tx();
637 furi_hal_infrared_tx_dma_set_polarity(0, INFRARED_POLARITY_SHIFT);
638 furi_hal_infrared_tx_dma_set_buffer(0);
639
640 furi_hal_infrared_state = InfraredStateAsyncTx;
641
642 LL_TIM_ClearFlag_UPDATE(INFRARED_DMA_TIMER);
643 LL_DMA_EnableChannel(INFRARED_DMA_CH1_DEF);
644 LL_DMA_EnableChannel(INFRARED_DMA_CH2_DEF);
645 furi_delay_us(5);
646 LL_TIM_GenerateEvent_UPDATE(INFRARED_DMA_TIMER); /* DMA -> TIMx_RCR */
647 furi_delay_us(5);
648 if(infrared_external_output) {
649 LL_GPIO_ResetOutputPin(
650 gpio_ext_pa7.port, gpio_ext_pa7.pin); /* when disable it prevents false pulse */
651 furi_hal_gpio_init_ex(
652 &gpio_ext_pa7, GpioModeAltFunctionPushPull, GpioPullUp, GpioSpeedHigh, GpioAltFn1TIM1);
653 } else {
654 LL_GPIO_ResetOutputPin(
655 gpio_infrared_tx.port,
656 gpio_infrared_tx.pin); /* when disable it prevents false pulse */
657 furi_hal_gpio_init_ex(
658 &gpio_infrared_tx,
659 GpioModeAltFunctionPushPull,
660 GpioPullUp,
661 GpioSpeedHigh,
662 GpioAltFn1TIM1);
663 }
664
665 FURI_CRITICAL_ENTER();
666 LL_TIM_GenerateEvent_UPDATE(INFRARED_DMA_TIMER); /* TIMx_RCR -> Repetition counter */
667 LL_TIM_EnableCounter(INFRARED_DMA_TIMER);
668 FURI_CRITICAL_EXIT();
669}
