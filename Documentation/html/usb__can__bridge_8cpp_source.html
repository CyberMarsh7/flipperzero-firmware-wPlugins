252static int32_t usb_can_tx_thread(void* context) {
253 UsbCanBridge* usb_can = (UsbCanBridge*)context;
254 uint8_t data[USB_CDC_PKT_LEN];
255 const char invalid_command[] = "[err]: invalid command received '%c' in ascii (0x%02X)\r\n";
256 while(1) {
257 uint32_t events =
258 furi_thread_flags_wait(WORKER_ALL_TX_EVENTS, FuriFlagWaitAny, FuriWaitForever);
259 furi_check(!(events & FuriFlagError));
260 if(events & WorkerEvtTxStop) break;
261 if(events & WorkerEvtCdcRx) {
262 WAIT_CDC();
263 size_t len = furi_hal_cdc_receive(usb_can->cfg.vcp_ch, data, USB_CDC_PKT_LEN);
264 cdc_busy = false;
265 if(usb_can->state == UsbCanLoopBackTestState) {
266 usb_can->st.tx_cnt += len;
267 WAIT_CDC();
268 furi_hal_cdc_send(usb_can->cfg.vcp_ch, data, len);
269
270 } else {
271 char* next_command = (char*)NULL;
272 uint8_t cmd_len = 0;
273 while(
274 usb_uart_bridge_get_next_command(&next_command, &cmd_len, (char*)data, len)) {
275 if(cmd_len > USB_CAN_BRIDGE_MAX_INPUT_LENGTH) {
276 M_USB_CAN_ERROR_INVALID_LENGTH();
277 //invalid input, stop processing
278 continue;
279 }
280
281 switch(next_command[0]) {
282 case '\n':
283 case '\r':
284 //ignore carriage return and line feed sent alone (ex. putty)=> this is not considered as an error
285 if(cmd_len != 1) {
286 M_USB_CAN_ERROR_INVALID_COMMAND();
287 }
288 break;
289 case 'S':
290 usb_can_bridge_configure(usb_can, (char*)next_command, cmd_len);
291 break;
292 case 'O':
293 case 'C':
294 usb_can_bridge_error(
295 usb_can, "[err]: just select app with the flipper buttons! \r\n");
296 break;
297 case 's':
298 case 'G':
299 case 'W':
300 case 'V':
301 case 'v':
302 case 'N':
303 case 'I':
304 case 'L':
305 case 'F':
306 case 'Z':
307 usb_can_bridge_error(
308 usb_can,
309 "[err]: command %c<x> not implemented yet!\r\n",
310 next_command[0]);
311 break;
312 case 'R':
313 usb_can_bridge_send(usb_can, next_command, cmd_len, true, true);
314 break;
315 case 'r':
316 usb_can_bridge_send(usb_can, next_command, cmd_len, false, true);
317 break;
318 case 'T':
319 usb_can_bridge_send(usb_can, next_command, cmd_len, true, false);
320 break;
321 case 't':
322 usb_can_bridge_send(usb_can, next_command, cmd_len, false, false);
323 break;
324 default:
325 M_USB_CAN_ERROR_INVALID_COMMAND();
326 break;
327 }
328 }
329 }
330 }
331 }
332 return 0;
333}

336static bool usb_uart_bridge_get_next_command(
337 char** next_command,
338 uint8_t* cmd_length,
339 char* cdc_data,
340 uint8_t cdc_frame_len) {
341 if(*next_command == NULL) {
342 *next_command = cdc_data;
343 } else {
344 //This is not first command : update command pointer
345 *next_command = &((*next_command)[*cmd_length]);
346 }
347 *cmd_length = 0;
348 while(((*next_command)[(*cmd_length)] != '\r') && ((*next_command)[(*cmd_length)] != '\n') &&
349 (!IS_LAST_CMD())) {
350 (*cmd_length)++;
351 }
352 if(IS_LAST_CMD()) {
353 return false;
354 } else {
355 (*cmd_length)++;
356 }
357 return true;
358}

387void usb_can_bridge_send(UsbCanBridge* usb_can, char* data, uint32_t len, bool extended, bool rtr) {
388 uint8_t data_bin[8];
389 uint32_t expected_len = 0;
390 uint32_t id;
391 uint32_t data_tmp;
392 uint32_t max_identifier;
393 uint8_t data_offset;
394 uint32_t min_length;
395 uint32_t max_length;
396 char frame_format[] = "t%3x%1x";
397 uint32_t dlc;
398
399 if(extended) {
400 max_identifier = 0x1FFFFFFF;
401 data_offset = 10;
402 if(rtr) {
403 frame_format[0] = 'R';
404 } else {
405 frame_format[0] = 'T';
406 }
407 frame_format[2] = '8';
408 } else {
409 max_identifier = 0x7FF;
410 data_offset = 5;
411 if(rtr) {
412 frame_format[0] = 'r';
413 };
414 }
415
416 min_length = data_offset;
417 max_length = min_length + 17;
418 if(!can_if_initialized) {
419 const char invalid_state[] = "[err]: please initialize CAN using Sx command!\r\n";
420 usb_can_bridge_error(usb_can, invalid_state, sizeof(invalid_state));
421 } else if(len >= min_length && len <= max_length) {
422 sscanf((const char*)&data[0], frame_format, (unsigned int*)&id, (unsigned int*)&dlc);
423 if(!rtr) {
424 expected_len = dlc;
425 }
426 if(id > max_identifier) {
427 const char invalid_identifier[] = "[err]: invalid can identifier (>%X): %X \r\n";
428 usb_can_bridge_error(usb_can, invalid_identifier, max_identifier, id);
429 } else if((len - data_offset) / 2 == expected_len) {
430 usb_can->can->mcpDigitalWrite(GPIO_PIN_0, GPIO_LOW);
431 usb_can->st.tx_cnt += expected_len;
432 for(uint32_t i = 0; i < expected_len; i++) {
433 sscanf((const char*)&data[data_offset + 2 * i], "%02lx", &data_tmp);
434 data_bin[i] = (uint8_t)(data_tmp & 0xFF);
435 }
436 usb_can->can->sendMsgBuf(id, extended, rtr, dlc, data_bin);
437
438 furi_timer_start(usb_can->timer, furi_kernel_get_tick_frequency() / 100);
439 } else {
440 usb_can_bridge_error(
441 usb_can,
442 "[err]: can payload length discrepency : expected %d got %d \r\n",
443 expected_len,
444 (len - data_offset) / 2);
445 }
446 } else {
447 M_USB_CAN_ERROR_INVALID_LENGTH();
448 }
449}
