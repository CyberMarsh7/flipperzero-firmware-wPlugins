298static int32_t usb_can_tx_thread(void* context) {
299 UsbCanBridge* usb_can = (UsbCanBridge*)context;
300 uint8_t data[USB_CDC_PKT_LEN];
301 const char invalid_command[] = "[err]: invalid command received '%c' in ascii (0x%02X)\r\n";
302 while(1) {
303 uint32_t events =
304 furi_thread_flags_wait(WORKER_ALL_TX_EVENTS, FuriFlagWaitAny, FuriWaitForever);
305 furi_check(!(events & FuriFlagError));
306 if(events & WorkerEvtTxStop) break;
307 if(events & WorkerEvtCdcRx) {
308 WAIT_CDC();
309 size_t len = furi_hal_cdc_receive(usb_can->cfg.vcp_ch, data, USB_CDC_PKT_LEN);
310 cdc_busy = false;
311 if(usb_can->state == UsbCanLoopBackTestState) {
312 usb_can->st.tx_cnt += len;
313 WAIT_CDC();
314 furi_hal_cdc_send(usb_can->cfg.vcp_ch, data, len);
315
316 } else {
317 char* next_command = (char*)NULL;
318 uint8_t cmd_len = 0;
319 while(
320 _usb_uart_bridge_get_next_command(&next_command, &cmd_len, (char*)data, len)) {
321 if(cmd_len > USB_CAN_BRIDGE_MAX_INPUT_LENGTH) {
322 M_USB_CAN_ERROR_INVALID_LENGTH();
323 //invalid input, stop processing
324 continue;
325 }
326
327 switch(next_command[0]) {
328 case '\n':
329 case '\r':
330 //ignore carriage return and line feed sent alone (ex. putty)=> this is not considered as an error
331 if(cmd_len != 1) {
332 M_USB_CAN_ERROR_INVALID_COMMAND();
333 }
334 break;
335 case 'S':
336 usb_can_bridge_configure(usb_can, (char*)next_command, cmd_len);
337 break;
338 case 'O':
339 case 'C':
340 usb_can_bridge_error(
341 usb_can, "[err]: just select app with the flipper buttons! \r\n");
342 break;
343 case 's':
344 case 'G':
345 case 'W':
346 case 'V':
347 case 'v':
348 case 'N':
349 case 'I':
350 case 'L':
351 case 'F':
352 case 'Z':
353 usb_can_bridge_error(
354 usb_can,
355 "[err]: command %c<x> not implemented yet!\r\n",
356 next_command[0]);
357 break;
358 case 'R':
359 usb_can_bridge_send(usb_can, next_command, cmd_len, true, true);
360 break;
361 case 'r':
362 usb_can_bridge_send(usb_can, next_command, cmd_len, false, true);
363 break;
364 case 'T':
365 usb_can_bridge_send(usb_can, next_command, cmd_len, true, false);
366 break;
367 case 't':
368 usb_can_bridge_send(usb_can, next_command, cmd_len, false, false);
369 break;
370 default:
371 M_USB_CAN_ERROR_INVALID_COMMAND();
372 break;
373 }
374 }
375 }
376 }
377 }
378 return 0;
379}

385#define IS_LAST_CMD() (&((*next_command)[(*cmd_length)]) == &cdc_data[cdc_frame_len])
406static bool _usb_uart_bridge_get_next_command(
407 char** next_command,
408 uint8_t* cmd_length,
409 char* cdc_data,
410 uint8_t cdc_frame_len) {
411 if(*next_command == NULL) {
412 *next_command = cdc_data;
413 } else {
414 //This is not first command : update command pointer
415 *next_command = &((*next_command)[*cmd_length]);
416 }
417 *cmd_length = 0;
418 while(((*next_command)[(*cmd_length)] != '\r') && ((*next_command)[(*cmd_length)] != '\n') &&
419 (!IS_LAST_CMD())) {
420 (*cmd_length)++;
421 }
422 if(IS_LAST_CMD()) {
423 return false;
424 } else {
425 (*cmd_length)++;
426 }
427 return true;
428}

499void usb_can_bridge_send(UsbCanBridge* usb_can, char* data, uint32_t len, bool extended, bool rtr) {
500 uint8_t data_bin[8];
501 uint32_t expected_len = 0;
502 uint32_t id;
503 uint32_t data_tmp;
504 uint32_t max_identifier;
505 uint8_t data_offset;
506 uint32_t min_length;
507 uint32_t max_length;
508 char frame_format[] = "t%3x%1x";
509 uint32_t dlc;
510
511 if(extended) {
512 max_identifier = 0x1FFFFFFF;
513 data_offset = 10;
514 if(rtr) {
515 frame_format[0] = 'R';
516 } else {
517 frame_format[0] = 'T';
518 }
519 frame_format[2] = '8';
520 } else {
521 max_identifier = 0x7FF;
522 data_offset = 5;
523 if(rtr) {
524 frame_format[0] = 'r';
525 };
526 }
527
528 min_length = data_offset;
529 max_length = min_length + 17;
530 if(!can_if_initialized) {
531 const char invalid_state[] = "[err]: please initialize CAN using Sx command!\r\n";
532 usb_can_bridge_error(usb_can, invalid_state, sizeof(invalid_state));
533 } else if(len >= min_length && len <= max_length) {
534 sscanf((const char*)&data[0], frame_format, (unsigned int*)&id, (unsigned int*)&dlc);
535 if(!rtr) {
536 expected_len = dlc;
537 }
538 if(id > max_identifier) {
539 const char invalid_identifier[] = "[err]: invalid can identifier (>%X): %X \r\n";
540 usb_can_bridge_error(usb_can, invalid_identifier, max_identifier, id);
541 } else if((len - data_offset) / 2 == expected_len) {
542 usb_can->can->mcpDigitalWrite(GPIO_PIN_0, GPIO_LOW);
543 usb_can->st.tx_cnt += expected_len;
544 for(uint32_t i = 0; i < expected_len; i++) {
545 sscanf((const char*)&data[data_offset + 2 * i], "%02lx", &data_tmp);
546 data_bin[i] = (uint8_t)(data_tmp & 0xFF);
547 }
548 usb_can->can->sendMsgBuf(id, extended, rtr, dlc, data_bin);
549
550 furi_timer_start(usb_can->timer, furi_kernel_get_tick_frequency() / 100);
551 } else {
552 usb_can_bridge_error(
553 usb_can,
554 "[err]: can payload length discrepency : expected %d got %d \r\n",
555 expected_len,
556 (len - data_offset) / 2);
557 }
558 } else {
559 M_USB_CAN_ERROR_INVALID_LENGTH();
560 }
561}
