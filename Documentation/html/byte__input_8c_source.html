118static char byte_input_get_nibble_text(uint8_t byte, bool high_nibble) {
119 if(high_nibble) {
120 byte = byte >> 4;
121 }
122 byte = byte & 0x0F;
123
124 switch(byte & 0x0F) {
125 case 0x0:
126 case 0x1:
127 case 0x2:
128 case 0x3:
129 case 0x4:
130 case 0x5:
131 case 0x6:
132 case 0x7:
133 case 0x8:
134 case 0x9:
135 byte = byte + '0';
136 break;
137 case 0xA:
138 case 0xB:
139 case 0xC:
140 case 0xD:
141 case 0xE:
142 case 0xF:
143 byte = byte - 0xA + 'A';
144 break;
145 default:
146 byte = '!';
147 break;
148 }
149
150 return byte;
151}

358static void byte_input_set_nibble(uint8_t* data, uint8_t position, char value, bool high_nibble) {
359 switch(value) {
360 case '0':

362 case '2':
363 case '3':
364 case '4':
365 case '5':
366 case '6':
367 case '7':
368 case '8':

730static bool byte_input_view_input_callback(InputEvent* event, void* context) {
731 ByteInput* byte_input = context;
732 furi_assert(byte_input);
733 bool consumed = false;
734
735 if(event->type == InputTypeShort || event->type == InputTypeRepeat) {
736 switch(event->key) {
737 case InputKeyLeft:
738 with_view_model(
739 byte_input->view, ByteInputModel * model, { byte_input_handle_left(model); }, true);
740 consumed = true;
741 break;
742 case InputKeyRight:
743 with_view_model(
744 byte_input->view,
745 ByteInputModel * model,
746 { byte_input_handle_right(model); },
747 true);
748 consumed = true;
749 break;
750 case InputKeyUp:
751 with_view_model(
752 byte_input->view, ByteInputModel * model, { byte_input_handle_up(model); }, true);
753 consumed = true;
754 break;
755 case InputKeyDown:
756 with_view_model(
757 byte_input->view, ByteInputModel * model, { byte_input_handle_down(model); }, true);
758 consumed = true;
759 break;
760 case InputKeyOk:
761 with_view_model(
762 byte_input->view, ByteInputModel * model, { byte_input_handle_ok(model); }, true);
763 consumed = true;
764 break;
765 default:
766 break;
767 }
768 }
769
770 if(event->type == InputTypeShort && event->key == InputKeyBack) {
771 // Back to keyboard
772 with_view_model(
773 byte_input->view,
774 ByteInputModel * model,
775 {
776 if(model->selected_row == -2) {
777 model->selected_row += 1;
778 consumed = true;
779 };
780 },
781 true);
782 }
783
784 if((event->type == InputTypeLong || event->type == InputTypeRepeat) &&
785 event->key == InputKeyBack) {
786 with_view_model(
787 byte_input->view,
788 ByteInputModel * model,
789 { byte_input_clear_selected_byte(model); },
790 true);
791 consumed = true;
792 }
793
794 return consumed;
795}
