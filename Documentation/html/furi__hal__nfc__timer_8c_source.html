114static int32_t furi_hal_nfc_timer_get_compensation(FuriHalNfcTimer timer) {
115 const FuriHalNfcTechBase* current_tech = furi_hal_nfc_tech[furi_hal_nfc.tech];
116
117 if(furi_hal_nfc.mode == FuriHalNfcModePoller) {
118 const FuriHalNfcPollerCompensation* comp = &current_tech->poller.compensation;
119 if(timer == FuriHalNfcTimerFwt)
120 return comp->fwt;
121 else if(timer == FuriHalNfcTimerBlockTx)
122 return comp->fdt;
123
124 } else if(furi_hal_nfc.mode == FuriHalNfcModeListener) {
125 const FuriHalNfcListenerCompensation* comp = &current_tech->listener.compensation;
126 if(timer == FuriHalNfcTimerBlockTx) return comp->fdt;
127 }
128
129 return 0;
130}

136static void furi_hal_nfc_timer_start_core_ticks(FuriHalNfcTimer timer, uint64_t core_ticks) {
137 furi_check(!furi_hal_nfc_timer_is_running(timer));
138
139 const FuriHalNfcTimerConfig* config = &furi_hal_nfc_timers[timer];
140 furi_check(furi_hal_bus_is_enabled(config->bus));
141
142 const uint32_t prescaler = (core_ticks - 1) / UINT16_MAX;
143 furi_check(prescaler <= UINT16_MAX);
144
145 const uint32_t arr_reg = core_ticks / (prescaler + 1);
146 furi_check(arr_reg <= UINT16_MAX);
147
148 LL_TIM_DisableIT_UPDATE(config->timer);
149
150 LL_TIM_SetPrescaler(config->timer, prescaler);
151 LL_TIM_SetAutoReload(config->timer, arr_reg);
152
153 LL_TIM_GenerateEvent_UPDATE(config->timer);
154 while(!LL_TIM_IsActiveFlag_UPDATE(config->timer))
155 ;
156 LL_TIM_ClearFlag_UPDATE(config->timer);
157
158 LL_TIM_EnableIT_UPDATE(config->timer);
159 LL_TIM_EnableCounter(config->timer);
160#ifdef FURI_HAL_NFC_TIMER_DEBUG
161 furi_hal_gpio_write(config->pin, true);
162#endif
163}

170static void furi_hal_nfc_timer_start_fc(FuriHalNfcTimer timer, uint32_t time_fc) {
171 const int32_t comp_fc = furi_hal_nfc_timer_get_compensation(timer);
172 // Not starting the timer if the compensation value is greater than the requested delay
173 if(comp_fc >= (int32_t)time_fc) return;
174
175 furi_hal_nfc_timer_start_core_ticks(
176 timer, ((uint64_t)SystemCoreClock * (time_fc - comp_fc)) / FURI_HAL_NFC_CARRIER_HZ);
177}
