80static int BrowserItem_t_cmp(const BrowserItem_t* a, const BrowserItem_t* b) {
81 // Back indicator comes before everything, then folders, then all other files.
82 if(a->type == BrowserItemTypeBack) {
83 return -1;
84 }
85 if(b->type == BrowserItemTypeBack) {
86 return 1;
87 }
88 if(xtreme_settings.sort_dirs_first) {
89 if(a->type == BrowserItemTypeFolder && b->type != BrowserItemTypeFolder) {
90 return -1;
91 }
92 if(a->type != BrowserItemTypeFolder && b->type == BrowserItemTypeFolder) {
93 return 1;
94 }
95 }
96
97 return furi_string_cmpi(a->display_name, b->display_name);

308 size_t array_size = items_array_size(model->items);

424static void
425 browser_list_item_cb(void* context, FuriString* item_path, bool is_folder, bool is_last) {
426 furi_assert(context);
427 FileBrowser* browser = (FileBrowser*)context;
428
429 BrowserItem_t item;
430 item.custom_icon_data = NULL;
431
432 if(!is_last) {
433 item.path = furi_string_alloc_set(item_path);
434 item.display_name = furi_string_alloc();
435 if(is_folder) {
436 item.type = BrowserItemTypeFolder;
437 } else {
438 item.type = BrowserItemTypeFile;
439 if(browser->item_callback) {
440 item.custom_icon_data = malloc(CUSTOM_ICON_MAX_SIZE);
441 if(!browser->item_callback(
442 item_path,
443 browser->item_context,
444 &item.custom_icon_data,
445 item.display_name)) {
446 free(item.custom_icon_data);
447 item.custom_icon_data = NULL;
448 }
449 }
450 }
451
452 if(furi_string_empty(item.display_name)) {
453 path_extract_filename(
454 item_path,
455 item.display_name,
456 (browser->hide_ext) && (item.type == BrowserItemTypeFile));
457 }
458
459 // We shouldn't update screen on each item if custom callback is not set
460 // Otherwise it will cause screen flickering
461 bool instant_update = (browser->item_callback != NULL);
462 with_view_model(
463 browser->view,
464 FileBrowserModel * model,
465 { items_array_push_back(model->items, item); },
466 instant_update);
467
468 furi_string_free(item.display_name);
469 furi_string_free(item.path);
470 if(item.custom_icon_data) {
471 free(item.custom_icon_data);
472 }
473 } else {
474 bool load_again = false;
475 int32_t load_offset = 0;
476 with_view_model(
477 browser->view,
478 FileBrowserModel * model,
479 {
480 if(model->item_cnt <= BROWSER_SORT_THRESHOLD) {
481 FuriString* selected = NULL;
482 if(model->item_idx > 0) {
483 selected = furi_string_alloc_set(
484 items_array_get(model->items, model->item_idx)->path);
485 }
486
487 items_array_sort(model->items);
488
489 if(selected != NULL) {
490 for(uint32_t i = 0; i < model->item_cnt; i++) {
491 if(!furi_string_cmp(items_array_get(model->items, i)->path, selected)) {
492 model->item_idx = i;
493 break;
494 }
495 }
496 }
497 }
498 if(browser_is_list_load_required(model)) {
499 model->list_loading = true;
500 load_again = true;
501 load_offset = CLAMP(
502 model->item_idx - ITEM_LIST_LEN_MAX / 2, (int32_t)model->item_cnt, 0);
503 } else {
504 model->list_loading = false;
505 }
506 },
507 true);
508 if(load_again) {
509 file_browser_worker_load(browser->worker, load_offset, ITEM_LIST_LEN_MAX);
510 } else {
511 browser_update_offset(browser);
512 }
513 }

652static bool file_browser_view_input_callback(InputEvent* event, void* context) {
653 FileBrowser* browser = context;
654 furi_assert(browser);
655 bool consumed = false;
656 bool is_loading = false;
657
658 with_view_model(
659 browser->view,
660 FileBrowserModel * model,
661 { is_loading = model->folder_loading || model->list_loading; },
662 false);
663
664 if(is_loading && event->key != InputKeyBack) {
665 return true; // Return without doing anything
666 } else if(event->key == InputKeyUp || event->key == InputKeyDown) {
667 if(event->type == InputTypeShort || event->type == InputTypeRepeat) {
668 with_view_model(
669 browser->view,
670 FileBrowserModel * model,
671 {
672 int32_t scroll_speed = 1;
673 if(model->button_held_for_ticks > 5) {
674 if(model->button_held_for_ticks % 2) {
675 scroll_speed = 0;
676 } else {
677 scroll_speed = model->button_held_for_ticks > 9 ? 5 : 3;
678 }
679 }
680 if(model->button_held_for_ticks < -1) {
681 model->button_held_for_ticks = 0;
682 }
683
684 if(event->key == InputKeyUp) {
685 if(model->item_idx < scroll_speed) {
686 // Would wrap around
687 if(model->item_idx == 0) {
688 // Is first item
689 if(model->button_held_for_ticks > 0) {
690 // Was holding, so wait a second to roll over
691 model->button_held_for_ticks = -1;
692 } else {
693 // Wasn't holding / done waiting, roll over now
694 model->item_idx = model->item_cnt - 1;
695 browser_list_rollover(model);
696 }
697 } else {
698 // Not first item, jump to first
699 model->item_idx = 0;
700 }
701 } else {
702 // No wrap around
703 model->item_idx =
704 ((model->item_idx - scroll_speed) + model->item_cnt) %
705 model->item_cnt;
706 }
707
708 if(browser_is_list_load_required(model)) {
709 model->list_loading = true;
710 int32_t load_offset = CLAMP(
711 model->item_idx - ITEM_LIST_LEN_MAX / 4 * 3,
712 (int32_t)model->item_cnt,
713 0);
714 file_browser_worker_load(
715 browser->worker, load_offset, ITEM_LIST_LEN_MAX);
716 }
717 model->scroll_counter = 0;
718
719 model->button_held_for_ticks += 1;
720 } else if(event->key == InputKeyDown) {
721 if(model->item_idx + scroll_speed >= (int32_t)model->item_cnt) {
722 // Would wrap around
723 if(model->item_idx == (int32_t)model->item_cnt - 1) {
724 // Is last item
725 if(model->button_held_for_ticks > 0) {
726 // Was holding, so wait a second to roll over
727 model->button_held_for_ticks = -1;
728 } else {
729 // Wasn't holding / done waiting, roll over now
730 model->item_idx = 0;
731 browser_list_rollover(model);
732 }
733 } else {
734 // Not last item, jump to last
735 model->item_idx = model->item_cnt - 1;
736 }
737 } else {
738 // No wrap around
739 model->item_idx = (model->item_idx + scroll_speed) % model->item_cnt;
740 }
741
742 if(browser_is_list_load_required(model)) {
743 model->list_loading = true;
744 int32_t load_offset = CLAMP(
745 model->item_idx - ITEM_LIST_LEN_MAX / 4 * 1,
746 (int32_t)model->item_cnt,
747 0);
748 file_browser_worker_load(
749 browser->worker, load_offset, ITEM_LIST_LEN_MAX);
750 }
751 model->scroll_counter = 0;
752
753 model->button_held_for_ticks += 1;
754 }
755 },
756 true);
757 browser_update_offset(browser);
758 consumed = true;
759 } else if(event->type == InputTypeRelease) {
760 with_view_model(
761 browser->view,
762 FileBrowserModel * model,
763 { model->button_held_for_ticks = 0; },
764 true);
765 }
766 } else if(event->key == InputKeyOk) {
767 if(event->type == InputTypeShort) {
768 BrowserItem_t* selected_item = NULL;
769 with_view_model(
770 browser->view,
771 FileBrowserModel * model,
772 {
773 if(browser_is_item_in_array(model, model->item_idx)) {
774 selected_item =
775 items_array_get(model->items, model->item_idx - model->array_offset);
776 }
777 },
778 false);
779
780 if(selected_item) {
781 if(selected_item->type == BrowserItemTypeBack) {
782 file_browser_worker_folder_exit(browser->worker);
783 } else if(selected_item->type == BrowserItemTypeFolder) {
784 file_browser_worker_folder_enter(browser->worker, selected_item->path, 0);
785 } else if(selected_item->type == BrowserItemTypeFile) {
786 furi_string_set(browser->result_path, selected_item->path);
787 if(browser->callback) {
788 browser->callback(browser->context);
789 }
790 }
791 }
792 consumed = true;
793 }
794 } else if(event->key == InputKeyLeft) {
795 if(event->type == InputTypeShort) {
796 bool is_root = false;
797 with_view_model(
798 browser->view, FileBrowserModel * model, { is_root = model->is_root; }, false);
799 if(!is_root) {
800 file_browser_worker_folder_exit(browser->worker);
801 }
802 consumed = true;
803 }
804 } else if(event->key == InputKeyBack) {
805 if(event->type == InputTypeShort) {
806 bool is_root = false;
807 with_view_model(
808 browser->view, FileBrowserModel * model, { is_root = model->is_root; }, false);
809
810 if(!is_root && !file_browser_worker_is_in_start_folder(browser->worker)) {
811 consumed = true;
812 file_browser_worker_folder_exit(browser->worker);
813 }
814 }
815 }
816
817 return consumed;
