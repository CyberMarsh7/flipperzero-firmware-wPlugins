128void furi_hal_resources_init_early() {
129 furi_hal_bus_enable(FuriHalBusGPIOA);
130 furi_hal_bus_enable(FuriHalBusGPIOB);
131 furi_hal_bus_enable(FuriHalBusGPIOC);
132 furi_hal_bus_enable(FuriHalBusGPIOD);
133 furi_hal_bus_enable(FuriHalBusGPIOE);
134 furi_hal_bus_enable(FuriHalBusGPIOH);
135
136 furi_hal_resources_init_input_pins(GpioModeInput);
137
138 // Explicit, surviving reset, pulls
139 LL_PWR_EnablePUPDCfg();
140 LL_PWR_EnableGPIOPullDown(LL_PWR_GPIO_A, LL_PWR_GPIO_BIT_8); // gpio_vibro
141 LL_PWR_EnableGPIOPullDown(LL_PWR_GPIO_B, LL_PWR_GPIO_BIT_8); // gpio_speaker
142 LL_PWR_EnableGPIOPullDown(LL_PWR_GPIO_B, LL_PWR_GPIO_BIT_9); // gpio_infrared_tx
143
144 // SD Card stepdown control
145 furi_hal_gpio_write(&gpio_periph_power, 1);
146 furi_hal_gpio_init(&gpio_periph_power, GpioModeOutputOpenDrain, GpioPullNo, GpioSpeedLow);
147
148 // Display pins
149 furi_hal_gpio_write(&gpio_display_rst_n, 0);
150 furi_hal_gpio_init_simple(&gpio_display_rst_n, GpioModeOutputPushPull);
151 LL_PWR_EnableGPIOPullUp(LL_PWR_GPIO_B, LL_PWR_GPIO_BIT_0); // gpio_display_rst_n
152 furi_hal_gpio_write(&gpio_display_di, 0);
153 furi_hal_gpio_init_simple(&gpio_display_di, GpioModeOutputPushPull);
154 LL_PWR_EnableGPIOPullDown(LL_PWR_GPIO_B, LL_PWR_GPIO_BIT_1); // gpio_display_di
155
156 // Hard reset USB
157 furi_hal_gpio_write(&gpio_usb_dm, 1);
158 furi_hal_gpio_write(&gpio_usb_dp, 1);
159 furi_hal_gpio_init_simple(&gpio_usb_dm, GpioModeOutputOpenDrain);
160 furi_hal_gpio_init_simple(&gpio_usb_dp, GpioModeOutputOpenDrain);
161 furi_hal_gpio_write(&gpio_usb_dm, 0);
162 furi_hal_gpio_write(&gpio_usb_dp, 0);
163 furi_delay_us(5); // Device Driven disconnect: 2.5us + extra to compensate cables
164 furi_hal_gpio_write(&gpio_usb_dm, 1);
165 furi_hal_gpio_write(&gpio_usb_dp, 1);
166 furi_hal_gpio_init_simple(&gpio_usb_dm, GpioModeAnalog);
167 furi_hal_gpio_init_simple(&gpio_usb_dp, GpioModeAnalog);
168 furi_hal_gpio_write(&gpio_usb_dm, 0);
169 furi_hal_gpio_write(&gpio_usb_dp, 0);
170
171 // External header pins
172 furi_hal_resources_init_gpio_pins(GpioModeAnalog);
173}
