93void furi_hal_version_set_name(const char* name) {
94 if(name != NULL && strlen(name)) {
95 strlcpy(furi_hal_version.name, name, FURI_HAL_VERSION_ARRAY_NAME_LENGTH);
96 snprintf(
97 furi_hal_version.device_name,
98 FURI_HAL_VERSION_DEVICE_NAME_LENGTH,
99 "x%s", // Someone tell me why that X is needed - it's for BLE adv name type (6 lines below)
100 furi_hal_version.name);
101 } else {
102 snprintf(furi_hal_version.device_name, FURI_HAL_VERSION_DEVICE_NAME_LENGTH, "xFlipper");
103 }
104
105 furi_hal_version.device_name[0] = AD_TYPE_COMPLETE_LOCAL_NAME;
106
107 // BLE Mac address
108 uint32_t udn = LL_FLASH_GetUDN();
109 if(version_get_custom_name(NULL) != NULL) {
110 udn = *((uint32_t*)version_get_custom_name(NULL));
111 }
112
113 uint32_t company_id = LL_FLASH_GetSTCompanyID();
114 // uint32_t device_id = LL_FLASH_GetDeviceID();
115 // Some flippers return 0x27 (flippers with chip revision 2003 6495) instead of 0x26 (flippers with chip revision 2001 6495)
116 // Mobile apps expects it to return 0x26
117 // Hardcoded here temporarily until mobile apps is updated to handle 0x27
118 uint32_t device_id = 0x26;
119 furi_hal_version.ble_mac[0] = (uint8_t)(udn & 0x000000FF);
120 furi_hal_version.ble_mac[1] = (uint8_t)((udn & 0x0000FF00) >> 8);
121 furi_hal_version.ble_mac[2] = (uint8_t)((udn & 0x00FF0000) >> 16);
122 furi_hal_version.ble_mac[3] = (uint8_t)device_id;
123 furi_hal_version.ble_mac[4] = (uint8_t)(company_id & 0x000000FF);
124 furi_hal_version.ble_mac[5] = (uint8_t)((company_id & 0x0000FF00) >> 8);
125}

165static void furi_hal_version_load_otp_v2() {
166 const FuriHalVersionOTPv2* otp = (FuriHalVersionOTPv2*)FURI_HAL_VERSION_OTP_ADDRESS;
167
168 // 1st block, programmed afer baking
169 furi_hal_version.timestamp = otp->header_timestamp;

171 // 2nd block, programmed afer baking
172 furi_hal_version.board_version = otp->board_version;

175 furi_hal_version.board_connect = otp->board_connect;
176 furi_hal_version.board_display = otp->board_display;
177
178 // 3rd and 4th blocks, programmed on FATP stage
179 if(otp->board_color != 0xFF) {

182 if(version_get_custom_name(NULL) != NULL) {

184 } else {
185 furi_hal_version_set_name(otp->name);

189 furi_hal_version.board_region = 0;

218FuriHalVersionOtpVersion furi_hal_version_get_otp_version() {
219 if(*(uint64_t*)FURI_HAL_VERSION_OTP_ADDRESS == 0xFFFFFFFF) {
220 return FuriHalVersionOtpVersionEmpty;
221 } else {
222 if(((FuriHalVersionOTPv1*)FURI_HAL_VERSION_OTP_ADDRESS)->header_magic ==
223 FURI_HAL_VERSION_OTP_HEADER_MAGIC) {
224 // Version 1+
225 uint8_t version = ((FuriHalVersionOTPv1*)FURI_HAL_VERSION_OTP_ADDRESS)->header_version;
226 if(version >= FuriHalVersionOtpVersion1 && version <= FuriHalVersionOtpVersion2) {
227 return version;
228 } else {
229 return FuriHalVersionOtpVersionUnknown;
230 }
231 } else if(((FuriHalVersionOTPv0*)FURI_HAL_VERSION_OTP_ADDRESS)->board_version <= 10) {
232 // Version 0
233 return FuriHalVersionOtpVersion0;
234 } else {
235 // Version Unknown
236 return FuriHalVersionOtpVersionUnknown;
237 }
238 }
