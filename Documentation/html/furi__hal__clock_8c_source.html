33void furi_hal_clock_init() {
34 /* HSE and HSI configuration and activation */
35 LL_RCC_HSE_SetCapacitorTuning(0x26);
36 LL_RCC_HSE_Enable();
37 LL_RCC_HSI_Enable();
38 while(!HS_CLOCK_IS_READY())
39 ;
40 /* Select HSI as system clock source after Wake Up from Stop mode
41 * Must be set before enabling CSS */
42 LL_RCC_SetClkAfterWakeFromStop(LL_RCC_STOP_WAKEUPCLOCK_HSI);
43
44 LL_RCC_HSE_EnableCSS();
45
46 /* LSE and LSI1 configuration and activation */
47 LL_PWR_EnableBkUpAccess();
48 LL_RCC_LSE_SetDriveCapability(LL_RCC_LSEDRIVE_HIGH);
49 LL_RCC_LSE_Enable();
50 LL_RCC_LSI1_Enable();
51 while(!LS_CLOCK_IS_READY())
52 ;
53
54 LL_EXTI_EnableIT_0_31(
55 LL_EXTI_LINE_18); /* Why? Because that's why. See RM0434, Table 61. CPU1 vector table. */
56 LL_EXTI_EnableRisingTrig_0_31(LL_EXTI_LINE_18);
57 LL_RCC_EnableIT_LSECSS();
58 /* ES0394, extended case of 2.2.2 */
59 if(!LL_RCC_IsActiveFlag_BORRST()) {
60 LL_RCC_LSE_EnableCSS();
61 }
62
63 /* Main PLL configuration and activation */
64 LL_RCC_PLL_ConfigDomain_SYS(LL_RCC_PLLSOURCE_HSE, LL_RCC_PLLM_DIV_2, 8, LL_RCC_PLLR_DIV_2);
65 LL_RCC_PLL_Enable();
66 LL_RCC_PLL_EnableDomain_SYS();
67 while(LL_RCC_PLL_IsReady() != 1)
68 ;
69
70 LL_RCC_PLLSAI1_ConfigDomain_48M(
71 LL_RCC_PLLSOURCE_HSE, LL_RCC_PLLM_DIV_2, 6, LL_RCC_PLLSAI1Q_DIV_2);
72 LL_RCC_PLLSAI1_ConfigDomain_ADC(
73 LL_RCC_PLLSOURCE_HSE, LL_RCC_PLLM_DIV_2, 6, LL_RCC_PLLSAI1R_DIV_2);
74 LL_RCC_PLLSAI1_Enable();
75 LL_RCC_PLLSAI1_EnableDomain_48M();
76 LL_RCC_PLLSAI1_EnableDomain_ADC();
77 while(LL_RCC_PLLSAI1_IsReady() != 1)
78 ;
79
80 /* Sysclk activation on the main PLL */
81 /* Set CPU1 prescaler */
82 LL_RCC_SetAHBPrescaler(LL_RCC_SYSCLK_DIV_1);
83
84 /* Set CPU2 prescaler, from this point we are not allowed to touch it. */
85 LL_C2_RCC_SetAHBPrescaler(LL_RCC_SYSCLK_DIV_2);
86
87 /* Prepare Flash memory for work on 64MHz system clock */
88 LL_FLASH_SetLatency(LL_FLASH_LATENCY_3);
89 while(LL_FLASH_GetLatency() != LL_FLASH_LATENCY_3)
90 ;
91
92 LL_RCC_SetSysClkSource(LL_RCC_SYS_CLKSOURCE_PLL);
93 while(LL_RCC_GetSysClkSource() != LL_RCC_SYS_CLKSOURCE_STATUS_PLL)
94 ;
95
96 /* Set AHB SHARED prescaler*/
97 LL_RCC_SetAHB4Prescaler(LL_RCC_SYSCLK_DIV_1);
98
99 /* Set APB1 prescaler*/
100 LL_RCC_SetAPB1Prescaler(LL_RCC_APB1_DIV_1);
101
102 /* Set APB2 prescaler*/
103 LL_RCC_SetAPB2Prescaler(LL_RCC_APB2_DIV_1);
104
105 /* Disable MSI */
106 LL_RCC_MSI_Disable();
107 while(LL_RCC_MSI_IsReady() != 0)
108 ;
109
110 /* Update CMSIS variable (which can be updated also through SystemCoreClockUpdate function) */
111 LL_SetSystemCoreClock(CPU_CLOCK_PLL_HZ);
112
113 /* Update the time base */
114 LL_Init1msTick(SystemCoreClock);
115 LL_SYSTICK_EnableIT();
116 NVIC_SetPriority(
117 SysTick_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), TICK_INT_PRIORITY, 0));
118 NVIC_EnableIRQ(SysTick_IRQn);
119
120 LL_RCC_SetCLK48ClockSource(LL_RCC_CLK48_CLKSOURCE_PLLSAI1);
121 LL_RCC_SetSMPSClockSource(LL_RCC_SMPS_CLKSOURCE_HSI);
122 LL_RCC_SetSMPSPrescaler(LL_RCC_SMPS_DIV_1);
123 LL_RCC_SetRFWKPClockSource(LL_RCC_RFWKP_CLKSOURCE_LSE);
124
125 FURI_LOG_I(TAG, "Init OK");
126}
