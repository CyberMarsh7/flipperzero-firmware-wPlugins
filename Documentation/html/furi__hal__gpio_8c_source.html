62void furi_hal_gpio_init_ex(
63 const GpioPin* gpio,
64 const GpioMode mode,
65 const GpioPull pull,
66 const GpioSpeed speed,
67 const GpioAltFn alt_fn) {
68 uint32_t sys_exti_port = GET_SYSCFG_EXTI_PORT(gpio->port);
69 uint32_t sys_exti_line = GET_SYSCFG_EXTI_LINE(gpio->pin);
70 uint32_t exti_line = GET_EXTI_LINE(gpio->pin);
71
72 // Configure gpio with interrupts disabled
73 FURI_CRITICAL_ENTER();
74
75 // Set gpio speed
76 switch(speed) {
77 case GpioSpeedLow:
78 LL_GPIO_SetPinSpeed(gpio->port, gpio->pin, LL_GPIO_SPEED_FREQ_LOW);
79 break;
80 case GpioSpeedMedium:
81 LL_GPIO_SetPinSpeed(gpio->port, gpio->pin, LL_GPIO_SPEED_FREQ_MEDIUM);
82 break;
83 case GpioSpeedHigh:
84 LL_GPIO_SetPinSpeed(gpio->port, gpio->pin, LL_GPIO_SPEED_FREQ_HIGH);
85 break;
86 case GpioSpeedVeryHigh:
87 LL_GPIO_SetPinSpeed(gpio->port, gpio->pin, LL_GPIO_SPEED_FREQ_VERY_HIGH);
88 break;
89 }
90
91 // Set gpio pull mode
92 switch(pull) {
93 case GpioPullNo:
94 LL_GPIO_SetPinPull(gpio->port, gpio->pin, LL_GPIO_PULL_NO);
95 break;
96 case GpioPullUp:
97 LL_GPIO_SetPinPull(gpio->port, gpio->pin, LL_GPIO_PULL_UP);
98 break;
99 case GpioPullDown:
100 LL_GPIO_SetPinPull(gpio->port, gpio->pin, LL_GPIO_PULL_DOWN);
101 break;
102 }
103
104 // Set gpio mode
105 if(mode >= GpioModeInterruptRise) {
106 // Set pin in interrupt mode
107 LL_GPIO_SetPinMode(gpio->port, gpio->pin, LL_GPIO_MODE_INPUT);
108 LL_SYSCFG_SetEXTISource(sys_exti_port, sys_exti_line);
109 if(mode == GpioModeInterruptRise || mode == GpioModeInterruptRiseFall) {
110 LL_EXTI_EnableIT_0_31(exti_line);
111 LL_EXTI_EnableRisingTrig_0_31(exti_line);
112 }
113 if(mode == GpioModeInterruptFall || mode == GpioModeInterruptRiseFall) {
114 LL_EXTI_EnableIT_0_31(exti_line);
115 LL_EXTI_EnableFallingTrig_0_31(exti_line);
116 }
117 if(mode == GpioModeEventRise || mode == GpioModeEventRiseFall) {
118 LL_EXTI_EnableEvent_0_31(exti_line);
119 LL_EXTI_EnableRisingTrig_0_31(exti_line);
120 }
121 if(mode == GpioModeEventFall || mode == GpioModeEventRiseFall) {
122 LL_EXTI_EnableEvent_0_31(exti_line);
123 LL_EXTI_EnableFallingTrig_0_31(exti_line);
124 }
125 } else {
126 // Disable interrupts if set
127 if(LL_SYSCFG_GetEXTISource(sys_exti_line) == sys_exti_port &&
128 LL_EXTI_IsEnabledIT_0_31(exti_line)) {
129 LL_EXTI_DisableIT_0_31(exti_line);
130 LL_EXTI_DisableRisingTrig_0_31(exti_line);
131 LL_EXTI_DisableFallingTrig_0_31(exti_line);
132 }
133
134 // Prepare alternative part if any
135 if(mode == GpioModeAltFunctionPushPull || mode == GpioModeAltFunctionOpenDrain) {
136 // set alternate function
137 if(furi_hal_gpio_get_pin_num(gpio) < 8) {
138 LL_GPIO_SetAFPin_0_7(gpio->port, gpio->pin, alt_fn);
139 } else {
140 LL_GPIO_SetAFPin_8_15(gpio->port, gpio->pin, alt_fn);
141 }
142 }
143
144 // Set not interrupt pin modes
145 switch(mode) {
146 case GpioModeInput:
147 LL_GPIO_SetPinMode(gpio->port, gpio->pin, LL_GPIO_MODE_INPUT);
148 break;
149 case GpioModeOutputPushPull:
150 LL_GPIO_SetPinOutputType(gpio->port, gpio->pin, LL_GPIO_OUTPUT_PUSHPULL);
151 LL_GPIO_SetPinMode(gpio->port, gpio->pin, LL_GPIO_MODE_OUTPUT);
152 break;
153 case GpioModeAltFunctionPushPull:
154 LL_GPIO_SetPinOutputType(gpio->port, gpio->pin, LL_GPIO_OUTPUT_PUSHPULL);
155 LL_GPIO_SetPinMode(gpio->port, gpio->pin, LL_GPIO_MODE_ALTERNATE);
156 break;
157 case GpioModeOutputOpenDrain:
158 LL_GPIO_SetPinOutputType(gpio->port, gpio->pin, LL_GPIO_OUTPUT_OPENDRAIN);
159 LL_GPIO_SetPinMode(gpio->port, gpio->pin, LL_GPIO_MODE_OUTPUT);
160 break;
161 case GpioModeAltFunctionOpenDrain:
162 LL_GPIO_SetPinOutputType(gpio->port, gpio->pin, LL_GPIO_OUTPUT_OPENDRAIN);
163 LL_GPIO_SetPinMode(gpio->port, gpio->pin, LL_GPIO_MODE_ALTERNATE);
164 break;
165 case GpioModeAnalog:
166 LL_GPIO_SetPinMode(gpio->port, gpio->pin, LL_GPIO_MODE_ANALOG);
167 break;
168 default:
169 break;
170 }
171 }
172 FURI_CRITICAL_EXIT();
173}

175void furi_hal_gpio_add_int_callback(const GpioPin* gpio, GpioExtiCallback cb, void* ctx) {
176 furi_assert(gpio);
177 furi_assert(cb);
178
179 FURI_CRITICAL_ENTER();
180 uint8_t pin_num = furi_hal_gpio_get_pin_num(gpio);
181 volatile GpioInterrupt* interrupt = &gpio_interrupt[pin_num];
182 furi_check(
183 (interrupt->callback == NULL) ||
184 ((interrupt->callback == cb) && (interrupt->context == ctx)));
185 interrupt->callback = cb;
186 interrupt->context = ctx;
187 interrupt->ready = true;
188 FURI_CRITICAL_EXIT();
189}
