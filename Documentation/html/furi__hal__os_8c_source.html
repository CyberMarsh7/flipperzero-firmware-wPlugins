135static inline uint32_t furi_hal_os_sleep(TickType_t expected_idle_ticks) {
136 // Stop ticks
137 furi_hal_clock_suspend_tick();
138
139 // Start wakeup timer
140 furi_hal_idle_timer_start(FURI_HAL_OS_TICKS_TO_IDLE_CNT(expected_idle_ticks));
141
142#ifdef FURI_HAL_OS_DEBUG
143 furi_hal_gpio_write(FURI_HAL_OS_DEBUG_AWAKE_GPIO, 0);
144#endif
145
146 // Go to sleep mode
147 furi_hal_power_sleep();
148
149#ifdef FURI_HAL_OS_DEBUG
150 furi_hal_gpio_write(FURI_HAL_OS_DEBUG_AWAKE_GPIO, 1);
151#endif
152
153 // Calculate how much time we spent in the sleep
154 uint32_t after_cnt = furi_hal_idle_timer_get_cnt() + furi_hal_os_skew + FURI_HAL_OS_EXTRA_CNT;
155 uint32_t after_tick = FURI_HAL_OS_IDLE_CNT_TO_TICKS(after_cnt);
156 furi_hal_os_skew = after_cnt - FURI_HAL_OS_TICKS_TO_IDLE_CNT(after_tick);
157
158 bool cmpm = LL_LPTIM_IsActiveFlag_CMPM(FURI_HAL_IDLE_TIMER);
159 bool arrm = LL_LPTIM_IsActiveFlag_ARRM(FURI_HAL_IDLE_TIMER);
160 if(cmpm && arrm) after_tick += expected_idle_ticks;
161
162 // Prepare tick timer for new round
163 furi_hal_idle_timer_reset();
164
165 // Resume ticks
166 furi_hal_clock_resume_tick();
167 return after_tick;
168}

170void vPortSuppressTicksAndSleep(TickType_t expected_idle_ticks) {
171 if(!furi_hal_power_sleep_available()) {
172 __WFI();
173 return;
174 }
175
176 // Core2 shenanigans takes extra time, so we want to compensate tick skew by reducing sleep duration by 1 tick
177 TickType_t unexpected_idle_ticks = expected_idle_ticks - 1;
178
179 // Limit amount of ticks to maximum that timer can count
180 if(unexpected_idle_ticks > FURI_HAL_OS_MAX_SLEEP) {
181 unexpected_idle_ticks = FURI_HAL_OS_MAX_SLEEP;
182 }
183
184 // Stop IRQ handling, no one should disturb us till we finish
185 __disable_irq();
186 do {
187 // Confirm OS that sleep is still possible
188 if(eTaskConfirmSleepModeStatus() == eAbortSleep || furi_hal_os_is_pending_irq()) {
189 break;
190 }
191
192 // Sleep and track how much ticks we spent sleeping
193 uint32_t completed_ticks = furi_hal_os_sleep(unexpected_idle_ticks);
194 // Notify system about time spent in sleep
195 if(completed_ticks > 0) {
196 if(completed_ticks > expected_idle_ticks) {
197#ifdef FURI_HAL_OS_DEBUG
198 furi_hal_console_printf(">%lu\r\n", completed_ticks - expected_idle_ticks);
199#endif
200 completed_ticks = expected_idle_ticks;
201 }
202 vTaskStepTick(completed_ticks);
203 }
204 } while(0);
205 // Reenable IRQ
206 __enable_irq();
207}
