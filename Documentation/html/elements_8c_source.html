233static size_t
234 elements_get_max_chars_to_fit(Canvas* canvas, Align horizontal, const char* text, uint8_t x) {
235 const char* end = strchr(text, '\n');
236 if(end == NULL) {
237 end = text + strlen(text);
238 }
239 size_t text_size = end - text;
240 FuriString* str;
241 str = furi_string_alloc_set(text);
242 furi_string_left(str, text_size);
243 size_t result = 0;
244
245 uint16_t len_px = canvas_string_width(canvas, furi_string_get_cstr(str));
246 uint8_t px_left = 0;
247 if(horizontal == AlignCenter) {
248 if(x > (canvas_width(canvas) / 2)) {
249 px_left = (canvas_width(canvas) - x) * 2;
250 } else {
251 px_left = x * 2;
252 }
253 } else if(horizontal == AlignLeft) {
254 px_left = canvas_width(canvas) - x;
255 } else if(horizontal == AlignRight) {
256 px_left = x;
257 } else {
258 furi_crash();
259 }
260
261 if(len_px > px_left) {
262 size_t excess_symbols_approximately =
263 ceilf((float)(len_px - px_left) / ((float)len_px / (float)text_size));
264 // reduce to 5 to be sure dash fit, and next line will be at least 5 symbols long
265 if(excess_symbols_approximately > 0) {
266 excess_symbols_approximately = MAX(excess_symbols_approximately, 5u);
267 result = text_size - excess_symbols_approximately - 1;
268 } else {
269 result = text_size;
270 }
271 } else {
272 result = text_size;
273 }
274
275 furi_string_free(str);
276 return result;
277}

279void elements_multiline_text_aligned(
280 Canvas* canvas,
281 uint8_t x,
282 uint8_t y,
283 Align horizontal,
284 Align vertical,
285 const char* text) {
286 furi_assert(canvas);
287 furi_assert(text);
288
289 uint8_t lines_count = 0;
290 uint8_t font_height = canvas_current_font_height(canvas);
291 FuriString* line;
292
293 /* go through text line by line and count lines */
294 for(const char* start = text; start[0];) {
295 size_t chars_fit = elements_get_max_chars_to_fit(canvas, horizontal, start, x);
296 ++lines_count;
297 start += chars_fit;
298 start += start[0] == '\n' ? 1 : 0;
299 }
300
301 if(vertical == AlignBottom) {
302 y -= font_height * (lines_count - 1);
303 } else if(vertical == AlignCenter) {
304 y -= (font_height * (lines_count - 1)) / 2;
305 }
306
307 /* go through text line by line and print them */
308 for(const char* start = text; start[0];) {
309 size_t chars_fit = elements_get_max_chars_to_fit(canvas, horizontal, start, x);
310
311 if((start[chars_fit] == '\n') || (start[chars_fit] == 0)) {
312 line = furi_string_alloc_printf("%.*s", chars_fit, start);
313 } else if((y + font_height) > canvas_height(canvas)) {
314 line = furi_string_alloc_printf("%.*s...\n", chars_fit, start);
315 } else {
316 chars_fit -= 1; // account for the dash
317 line = furi_string_alloc_printf("%.*s-\n", chars_fit, start);
318 }
319 canvas_draw_str_aligned(canvas, x, y, horizontal, vertical, furi_string_get_cstr(line));
320 furi_string_free(line);
321 y += font_height;
322 if(y > canvas_height(canvas)) {
323 break;
324 }
325
326 start += chars_fit;
327 start += start[0] == '\n' ? 1 : 0;
328 }
329}

600void elements_scrollable_text_line_str(
601 Canvas* canvas,
602 uint8_t x,
603 uint8_t y,
604 uint8_t width,
605 const char* string,
606 size_t scroll,
607 bool ellipsis,
608 bool centered) {
609 FuriString* line = furi_string_alloc_set_str(string);
610
611 size_t len_px = canvas_string_width(canvas, furi_string_get_cstr(line));
612 if(len_px > width) {
613 if(centered) {
614 centered = false;
615 x -= width / 2;
616 }
617
618 if(ellipsis) {
619 width -= canvas_string_width(canvas, "...");
620 }
621
622 // Calculate scroll size
623 size_t scroll_size = furi_string_size(line);
624 size_t right_width = 0;
625 for(size_t i = scroll_size - 1; i > 0; i--) {
626 right_width += canvas_glyph_width(canvas, furi_string_get_char(line, i));
627 if(right_width > width) break;
628 scroll_size--;
629 if(!scroll_size) break;
630 }
631 // Ensure that we have something to scroll
632 if(scroll_size) {
633 scroll_size += 3;
634 scroll = scroll % scroll_size;
635 furi_string_right(line, scroll);
636 }
637
638 len_px = canvas_string_width(canvas, furi_string_get_cstr(line));
639 while(len_px > width) {
640 furi_string_left(line, furi_string_size(line) - 1);
641 len_px = canvas_string_width(canvas, furi_string_get_cstr(line));
642 }
643
644 if(ellipsis) {
645 furi_string_cat(line, "...");
646 }
647 }
648
649 if(centered) {
650 canvas_draw_str_aligned(
651 canvas, x, y, AlignCenter, AlignBottom, furi_string_get_cstr(line));
652 } else {
653 canvas_draw_str(canvas, x, y, furi_string_get_cstr(line));
654 }
655 furi_string_free(line);

683void elements_text_box(
684 Canvas* canvas,
685 uint8_t x,
686 uint8_t y,
687 uint8_t width,
688 uint8_t height,
689 Align horizontal,
690 Align vertical,
691 const char* text,
692 bool strip_to_dots) {
693 furi_assert(canvas);
694
695 ElementTextBoxLine line[ELEMENTS_MAX_LINES_NUM];
696 bool bold = false;
697 bool mono = false;
698 bool inverse = false;
699 bool inverse_present = false;
700 Font current_font = FontSecondary;
701 Font prev_font = FontSecondary;
702 const CanvasFontParameters* font_params = canvas_get_font_params(canvas, current_font);
703
704 // Fill line parameters
705 uint8_t line_leading_min = font_params->leading_min;
706 uint8_t line_leading_default = font_params->leading_default;
707 uint8_t line_height = font_params->height;
708 uint8_t line_descender = font_params->descender;
709 uint8_t line_num = 0;
710 uint8_t line_width = 0;
711 uint8_t line_len = 0;
712 uint8_t total_height_min = 0;
713 uint8_t total_height_default = 0;
714 uint16_t i = 0;
715 bool full_text_processed = false;
716 uint16_t dots_width = canvas_string_width(canvas, "...");
717
718 canvas_set_font(canvas, FontSecondary);
719
720 // Fill all lines
721 line[0].text = text;
722 for(i = 0; !full_text_processed; i++) {
723 line_len++;
724 // Identify line height
725 if(prev_font != current_font) {
726 font_params = canvas_get_font_params(canvas, current_font);
727 line_leading_min = MAX(line_leading_min, font_params->leading_min);
728 line_leading_default = MAX(line_leading_default, font_params->leading_default);
729 line_height = MAX(line_height, font_params->height);
730 line_descender = MAX(line_descender, font_params->descender);
731 prev_font = current_font;
732 }
733 // Set the font
734 if(text[i] == '\e' && text[i + 1]) {
735 i++;
736 line_len++;
737 if(text[i] == ELEMENTS_BOLD_MARKER) {
738 if(bold) {
739 current_font = FontSecondary;
740 } else {
741 current_font = FontPrimary;
742 }
743 canvas_set_font(canvas, current_font);
744 bold = !bold;
745 }
746 if(text[i] == ELEMENTS_MONO_MARKER) {
747 if(mono) {
748 current_font = FontSecondary;
749 } else {
750 current_font = FontKeyboard;
751 }
752 canvas_set_font(canvas, FontKeyboard);
753 mono = !mono;
754 }
755 if(text[i] == ELEMENTS_INVERSE_MARKER) {
756 inverse_present = true;
757 }
758 continue;
759 }
760 if(text[i] != '\n') {
761 line_width += canvas_glyph_width(canvas, text[i]);
762 }
763 // Process new line
764 if(text[i] == '\n' || text[i] == '\0' || line_width > width) {
765 if(line_width > width) {
766 line_width -= canvas_glyph_width(canvas, text[i--]);
767 line_len--;
768 }
769 if(text[i] == '\0') {
770 full_text_processed = true;
771 }
772 if(inverse_present) {
773 line_leading_min += 1;
774 line_leading_default += 1;
775 inverse_present = false;
776 }
777 line[line_num].leading_min = line_leading_min;
778 line[line_num].leading_default = line_leading_default;
779 line[line_num].height = line_height;
780 line[line_num].descender = line_descender;
781 if(total_height_min + line_leading_min > height) {
782 break;
783 }
784 total_height_min += line_leading_min;
785 total_height_default += line_leading_default;
786 line[line_num].len = line_len;
787 if(horizontal == AlignCenter) {
788 line[line_num].x = x + (width - line_width) / 2;
789 } else if(horizontal == AlignRight) {
790 line[line_num].x = x + (width - line_width);
791 } else {
792 line[line_num].x = x;
793 }
794 line[line_num].y = total_height_min;
795 line_num++;
796 if(text[i + 1]) {
797 line[line_num].text = &text[i + 1];
798 }
799
800 line_leading_min = font_params->leading_min;
801 line_height = font_params->height;
802 line_descender = font_params->descender;
803 line_width = 0;
804 line_len = 0;
805 }
806 }
807
808 // Set vertical alignment for all lines
809 if(total_height_default < height) {
810 if(vertical == AlignTop) {
811 line[0].y = y + line[0].height;
812 } else if(vertical == AlignCenter) {
813 line[0].y = y + line[0].height + (height - total_height_default) / 2;
814 } else if(vertical == AlignBottom) {
815 line[0].y = y + line[0].height + (height - total_height_default);
816 }
817 if(line_num > 1) {
818 for(uint8_t i = 1; i < line_num; i++) {
819 line[i].y = line[i - 1].y + line[i - 1].leading_default;
820 }
821 }
822 } else if(line_num > 1) {
823 uint8_t free_pixel_num = height - total_height_min;
824 uint8_t fill_pixel = 0;
825 uint8_t j = 1;
826 line[0].y = y + line[0].height;
827 while(fill_pixel < free_pixel_num) {
828 line[j].y = line[j - 1].y + line[j - 1].leading_min + 1;
829 fill_pixel++;
830 j = j % (line_num - 1) + 1;
831 }
832 }
833
834 // Draw line by line
835 canvas_set_font(canvas, FontSecondary);
836 bold = false;
837 mono = false;
838 inverse = false;
839 for(uint8_t i = 0; i < line_num; i++) {
840 for(uint8_t j = 0; j < line[i].len; j++) {
841 // Process format symbols
842 if(line[i].text[j] == ELEMENTS_BOLD_MARKER) {
843 if(bold) {
844 current_font = FontSecondary;
845 } else {
846 current_font = FontPrimary;
847 }
848 canvas_set_font(canvas, current_font);
849 bold = !bold;
850 continue;
851 }
852 if(line[i].text[j] == ELEMENTS_MONO_MARKER) {
853 if(mono) {
854 current_font = FontSecondary;
855 } else {
856 current_font = FontKeyboard;
857 }
858 canvas_set_font(canvas, current_font);
859 mono = !mono;
860 continue;
861 }
862 if(line[i].text[j] == ELEMENTS_INVERSE_MARKER) {
863 inverse = !inverse;
864 continue;
865 }
866 if(inverse) {
867 canvas_draw_box(
868 canvas,
869 line[i].x - 1,
870 line[i].y - line[i].height - 1,
871 canvas_glyph_width(canvas, line[i].text[j]) + 1,
872 line[i].height + line[i].descender + 2);
873 canvas_invert_color(canvas);
874 canvas_draw_glyph(canvas, line[i].x, line[i].y, line[i].text[j]);
875 canvas_invert_color(canvas);
876 } else {
877 if((i == line_num - 1) && strip_to_dots) {
878 uint8_t next_symbol_width = canvas_glyph_width(canvas, line[i].text[j]);
879 if(line[i].x + next_symbol_width + dots_width > x + width) {
880 canvas_draw_str(canvas, line[i].x, line[i].y, "...");
881 break;
882 }
883 }
884 canvas_draw_glyph(canvas, line[i].x, line[i].y, line[i].text[j]);
885 }
886 line[i].x += canvas_glyph_width(canvas, line[i].text[j]);
887 }
888 }
889 canvas_set_font(canvas, FontSecondary);
890}
