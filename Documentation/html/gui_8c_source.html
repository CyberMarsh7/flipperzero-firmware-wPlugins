310static void gui_input(Gui* gui, InputEvent* input_event) {
311 furi_assert(gui);
312 furi_assert(input_event);
313
314 // Check input complementarity
315 uint8_t key_bit = (1 << input_event->key);
316 if(input_event->type == InputTypeRelease) {
317 gui->ongoing_input &= ~key_bit;
318 } else if(input_event->type == InputTypePress) {
319 gui->ongoing_input |= key_bit;
320 } else if(!(gui->ongoing_input & key_bit)) {
321 FURI_LOG_D(
322 TAG,
323 "non-complementary input, discarding key: %s type: %s, sequence: %p",
324 input_get_key_name(input_event->key),
325 input_get_type_name(input_event->type),
326 (void*)input_event->sequence);
327 return;
328 }
329
330 gui_lock(gui);
331
332 do {
333 if(gui->direct_draw && !gui->ongoing_input_view_port) {
334 break;
335 }
336
337 ViewPort* view_port = NULL;
338
339 if(gui->lockdown) {
340 view_port = gui_view_port_find_enabled(gui->layers[GuiLayerDesktop]);
341 } else {
342 view_port = gui_view_port_find_enabled(gui->layers[GuiLayerFullscreen]);
343 if(!view_port) view_port = gui_view_port_find_enabled(gui->layers[GuiLayerWindow]);
344 if(!view_port) view_port = gui_view_port_find_enabled(gui->layers[GuiLayerDesktop]);
345 }
346
347 if(!(gui->ongoing_input & ~key_bit) && input_event->type == InputTypePress) {
348 gui->ongoing_input_view_port = view_port;
349 }
350
351 if(view_port && view_port == gui->ongoing_input_view_port) {
352 view_port_input(view_port, input_event);
353 } else if(gui->ongoing_input_view_port && input_event->type == InputTypeRelease) {
354 FURI_LOG_D(
355 TAG,
356 "ViewPort changed while key press %p -> %p. Sending key: %s, type: %s, sequence: %p to previous view port",
357 gui->ongoing_input_view_port,
358 view_port,
359 input_get_key_name(input_event->key),
360 input_get_type_name(input_event->type),
361 (void*)input_event->sequence);
362 view_port_input(gui->ongoing_input_view_port, input_event);
363 } else {
364 FURI_LOG_D(
365 TAG,
366 "ViewPort changed while key press %p -> %p. Discarding key: %s, type: %s, sequence: %p",
367 gui->ongoing_input_view_port,
368 view_port,
369 input_get_key_name(input_event->key),
370 input_get_type_name(input_event->type),
371 (void*)input_event->sequence);
372 }
373 } while(false);
374
375 gui_unlock(gui);
376}

583Gui* gui_alloc() {
584 Gui* gui = malloc(sizeof(Gui));
585 // Thread ID
586 gui->thread_id = furi_thread_get_current_id();
587 // Allocate mutex
588 gui->mutex = furi_mutex_alloc(FuriMutexTypeNormal);
589 furi_check(gui->mutex);
590 // Layers
591 for(size_t i = 0; i < GuiLayerMAX; i++) {
592 ViewPortArray_init(gui->layers[i]);
593 }
594 // Drawing canvas
595 gui->canvas = canvas_init();
596 CanvasCallbackPairArray_init(gui->canvas_callback_pair);
597
598 // Input
599 gui->input_queue = furi_message_queue_alloc(8, sizeof(InputEvent));
600 gui->input_events = furi_record_open(RECORD_INPUT_EVENTS);
601
602 furi_check(gui->input_events);
603 furi_pubsub_subscribe(gui->input_events, gui_input_events_callback, gui);
604
605 Storage* storage = furi_record_open(RECORD_STORAGE);
606 gui_add_view_port(gui, storage->sd_gui.view_port, GuiLayerStatusBarLeft);
607 furi_record_close(RECORD_STORAGE);
608
609 return gui;
610}

612int32_t gui_srv(void* p) {
613 UNUSED(p);
614 Gui* gui = gui_alloc();
615
616 furi_record_create(RECORD_GUI, gui);
617
618 while(1) {
619 uint32_t flags =
620 furi_thread_flags_wait(GUI_THREAD_FLAG_ALL, FuriFlagWaitAny, FuriWaitForever);
621 // Process and dispatch input
622 if(flags & GUI_THREAD_FLAG_INPUT) {
623 // Process till queue become empty
624 InputEvent input_event;
625 while(furi_message_queue_get(gui->input_queue, &input_event, 0) == FuriStatusOk) {
626 gui_input(gui, &input_event);
627 }
628 }
629 // Process and dispatch draw call
630 if(flags & GUI_THREAD_FLAG_DRAW) {
631 // Clear flags that arrived on input step
632 furi_thread_flags_clear(GUI_THREAD_FLAG_DRAW);
633 gui_redraw(gui);
634 }
635 }
636
637 return 0;
638}
