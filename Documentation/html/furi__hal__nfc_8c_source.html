350 furi_assert(tech < FuriHalNfcTechNum);
351
352 FuriHalSpiBusHandle* handle = &furi_hal_spi_bus_handle_nfc;
353
354 FuriHalNfcError error = FuriHalNfcErrorNone;

369 }
370
371 furi_hal_nfc.mode = mode;
372 furi_hal_nfc.tech = tech;
373 return error;
