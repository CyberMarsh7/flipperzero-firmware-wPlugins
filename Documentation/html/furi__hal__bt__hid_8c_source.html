193void furi_hal_bt_hid_start() {
194 // Start device info
195 if(!dev_info_svc_is_started()) {
196 dev_info_svc_start();
197 }
198 // Start battery service
199 if(!battery_svc_is_started()) {
200 battery_svc_start();
201 }
202 // Start HID service
203 if(!hid_svc_is_started()) {
204 hid_svc_start();
205 }
206 // Configure HID Keyboard
207 hid_svc_register_led_state_callback(furi_hal_bt_hid_led_state_cb, &hid_host_led_state);
208
209 kb_report = malloc(sizeof(FuriHalBtHidKbReport));
210 mouse_report = malloc(sizeof(FuriHalBtHidMouseReport));
211 consumer_report = malloc(sizeof(FuriHalBtHidConsumerReport));
212
213 // Configure Report Map characteristic
214 hid_svc_update_report_map(
215 furi_hal_bt_hid_report_map_data, sizeof(furi_hal_bt_hid_report_map_data));
216 // Configure HID Information characteristic
217 uint8_t hid_info_val[4] = {
218 FURI_HAL_BT_INFO_BASE_USB_SPECIFICATION & 0x00ff,
219 (FURI_HAL_BT_INFO_BASE_USB_SPECIFICATION & 0xff00) >> 8,
220 FURI_HAL_BT_INFO_COUNTRY_CODE,
221 FURI_HAL_BT_HID_INFO_FLAG_REMOTE_WAKE_MSK |
222 FURI_HAL_BT_HID_INFO_FLAG_NORMALLY_CONNECTABLE_MSK,
223 };
224 hid_svc_update_info(hid_info_val);
225}

252bool furi_hal_bt_hid_kb_press(uint16_t button) {
253 furi_assert(kb_report);
254 uint8_t i;
255 for(i = 0; i < FURI_HAL_BT_HID_KB_MAX_KEYS; i++) {
256 if(kb_report->key[i] == 0) {
257 kb_report->key[i] = button & 0xFF;
258 break;
259 }
260 }
261 if(i == FURI_HAL_BT_HID_KB_MAX_KEYS) {
262 return false;
263 }
264 kb_report->mods |= (button >> 8);
265 return hid_svc_update_input_report(
266 ReportNumberKeyboard, (uint8_t*)kb_report, sizeof(FuriHalBtHidKbReport));
267}

272 if(kb_report->key[i] == (button & 0xFF)) {
273 kb_report->key[i] = 0;
274 break;
275 }
276 }
277 kb_report->mods &= ~(button >> 8);
