65static bool
66 furi_hal_i2c_wait_for_end(I2C_TypeDef* i2c, FuriHalI2cEnd end, FuriHalCortexTimer timer) {
67 // If ending the transaction with a stop condition, wait for it to be detected, otherwise wait for a transfer complete flag
68 bool wait_for_stop = end == FuriHalI2cEndStop;
69 uint32_t end_mask = (wait_for_stop) ? I2C_ISR_STOPF : (I2C_ISR_TC | I2C_ISR_TCR);
70
71 while((i2c->ISR & end_mask) == 0) {
72 if(furi_hal_cortex_timer_is_expired(timer)) {
73 return false;
74 }
75 }
76
77 return true;
78}

116static bool furi_hal_i2c_transfer(
117 I2C_TypeDef* i2c,
118 uint8_t* data,
119 uint32_t size,
120 FuriHalI2cEnd end,
121 bool read,
122 FuriHalCortexTimer timer) {
123 bool ret = true;
124
125 while(size > 0) {
126 bool should_stop = furi_hal_cortex_timer_is_expired(timer) ||
127 furi_hal_i2c_transfer_is_aborted(i2c);
128
129 // Modifying the data pointer's data is UB if read is true
130 if(read && LL_I2C_IsActiveFlag_RXNE(i2c)) {
131 *data = LL_I2C_ReceiveData8(i2c);
132 data++;
133 size--;
134 } else if(!read && LL_I2C_IsActiveFlag_TXIS(i2c)) {
135 LL_I2C_TransmitData8(i2c, *data);
136 data++;
137 size--;
138 }
139
140 // Exit on timeout or premature stop, probably caused by a nacked address or byte
141 if(should_stop) {
142 ret = size == 0; // If the transfer was over, still a success
143 break;
144 }
145 }
146
147 if(ret) {
148 ret = furi_hal_i2c_wait_for_end(i2c, end, timer);
149 }
150
151 LL_I2C_ClearFlag_STOP(i2c);
152
153 return ret;
154}

156static bool furi_hal_i2c_transaction(
157 I2C_TypeDef* i2c,
158 uint16_t address,
159 bool ten_bit,
160 uint8_t* data,
161 size_t size,
162 FuriHalI2cBegin begin,
163 FuriHalI2cEnd end,
164 bool read,
165 FuriHalCortexTimer timer) {
166 uint32_t addr_size = ten_bit ? LL_I2C_ADDRSLAVE_10BIT : LL_I2C_ADDRSLAVE_7BIT;
167 uint32_t start_signal = furi_hal_i2c_get_start_signal(begin, ten_bit, read);
168
169 if(!furi_hal_i2c_wait_for_idle(i2c, begin, timer)) {
170 return false;
171 }
172
173 do {
174 uint8_t transfer_size = size;
175 FuriHalI2cEnd transfer_end = end;
176
177 if(size > 255) {
178 transfer_size = 255;
179 transfer_end = FuriHalI2cEndPause;
180 }
181
182 uint32_t end_signal = furi_hal_i2c_get_end_signal(transfer_end);
183
184 LL_I2C_HandleTransfer(i2c, address, addr_size, transfer_size, end_signal, start_signal);
185
186 if(!furi_hal_i2c_transfer(i2c, data, transfer_size, transfer_end, read, timer)) {
187 return false;
188 }
189
190 size -= transfer_size;
191 data += transfer_size;
192 start_signal = LL_I2C_GENERATE_NOSTARTSTOP;
193 } while(size > 0);
194
195 return true;
196}

284bool furi_hal_i2c_is_device_ready(FuriHalI2cBusHandle* handle, uint8_t i2c_addr, uint32_t timeout) {
285 furi_check(handle);
286 furi_check(handle->bus->current_handle == handle);
287 furi_assert(timeout > 0);
288
289 bool ret = true;
290 FuriHalCortexTimer timer = furi_hal_cortex_timer_get(timeout * 1000);
291

298 i2c_addr,

302 LL_I2C_GENERATE_START_WRITE);
303

311 LL_I2C_ClearFlag_STOP(handle->bus->i2c);
