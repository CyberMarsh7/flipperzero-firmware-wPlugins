155bool furi_hal_bt_start_radio_stack() {
156 bool res = false;
157 furi_assert(furi_hal_bt.core2_mtx);
158
159 furi_mutex_acquire(furi_hal_bt.core2_mtx, FuriWaitForever);
160
161 // Explicitly tell that we are in charge of CLK48 domain
162 furi_check(LL_HSEM_1StepLock(HSEM, CFG_HW_CLK48_CONFIG_SEMID) == 0);
163
164 do {
165 // Wait until C2 is started or timeout
166 if(!ble_glue_wait_for_c2_start(FURI_HAL_BT_C2_START_TIMEOUT)) {
167 FURI_LOG_E(TAG, "Core2 start failed");
168 ble_glue_thread_stop();
169 break;
170 }
171
172 // If C2 is running, start radio stack fw
173 if(!furi_hal_bt_ensure_c2_mode(BleGlueC2ModeStack)) {
174 break;
175 }
176
177 // Check whether we support radio stack
178 const BleGlueC2Info* c2_info = ble_glue_get_c2_info();
179 if(!furi_hal_bt_radio_stack_is_supported(c2_info)) {
180 FURI_LOG_E(TAG, "Unsupported radio stack");
181 // Don't stop SHCI for crypto enclave support
182 break;
183 }
184 // Starting radio stack
185 if(!ble_glue_start()) {
186 FURI_LOG_E(TAG, "Failed to start radio stack");
187 ble_glue_thread_stop();
188 ble_app_thread_stop();
189 break;
190 }
191 res = true;
192 } while(false);
193 furi_mutex_release(furi_hal_bt.core2_mtx);
194
195 return res;

218bool furi_hal_bt_start_app(FuriHalBtProfile profile, GapEventCallback event_cb, void* context) {
219 furi_assert(event_cb);
220 furi_assert(profile < FuriHalBtProfileNumber);
221 bool ret = false;
222
223 do {
224 if(!ble_glue_is_radio_stack_ready()) {
225 FURI_LOG_E(TAG, "Can't start BLE App - radio stack did not start");
226 break;
227 }
228 if(!furi_hal_bt_is_ble_gatt_gap_supported()) {
229 FURI_LOG_E(TAG, "Can't start Ble App - unsupported radio stack");
230 break;
231 }
232 GapConfig* config = &profile_config[profile].config;
233 // Configure GAP
234 if(profile == FuriHalBtProfileSerial) {
235 // Set mac address
236 memcpy(
237 config->mac_address, furi_hal_version_get_ble_mac(), sizeof(config->mac_address));
238 // Set advertise name
239 strlcpy(
240 config->adv_name,
241 furi_hal_version_get_ble_local_device_name_ptr(),
242 FURI_HAL_VERSION_DEVICE_NAME_LENGTH);
243
244 config->adv_service_uuid |= furi_hal_version_get_hw_color();
245 } else if(profile == FuriHalBtProfileHidKeyboard) {
246 // Change MAC address for HID profile
247 const uint8_t* normal_mac = furi_hal_version_get_ble_mac();
248 uint8_t empty_mac[sizeof(config->mac_address)] = FURI_HAL_BT_EMPTY_MAC_ADDR;
249 uint8_t default_mac[sizeof(config->mac_address)] = FURI_HAL_BT_DEFAULT_MAC_ADDR;
250 if(memcmp(config->mac_address, empty_mac, sizeof(config->mac_address)) == 0 ||
251 memcmp(config->mac_address, normal_mac, sizeof(config->mac_address)) == 0 ||
252 memcmp(config->mac_address, default_mac, sizeof(config->mac_address)) == 0) {
253 memcpy(config->mac_address, normal_mac, sizeof(config->mac_address));
254 config->mac_address[2]++;
255 }
256 // Change name Flipper -> Control
257 if(strnlen(config->adv_name, FURI_HAL_VERSION_DEVICE_NAME_LENGTH) < 2 ||
258 strnlen(config->adv_name + 1, FURI_HAL_VERSION_DEVICE_NAME_LENGTH - 1) < 1) {
259 snprintf(
260 config->adv_name,
261 FURI_HAL_VERSION_DEVICE_NAME_LENGTH,
262 "%cControl %s",
263 AD_TYPE_COMPLETE_LOCAL_NAME,
264 furi_hal_version_get_name_ptr());
265 }
266 }
267 if(!gap_init(config, event_cb, context)) {
268 gap_thread_stop();
269 FURI_LOG_E(TAG, "Failed to init GAP");
270 break;
271 }
272 // Start selected profile services
273 if(furi_hal_bt_is_ble_gatt_gap_supported()) {
274 profile_config[profile].start();
275 }
276 ret = true;
277 } while(false);
278 current_profile = &profile_config[profile];
279
280 return ret;
281}

453float furi_hal_bt_get_rssi() {
454 float val;
455 uint8_t rssi_raw[3];
456
457 if(aci_hal_read_raw_rssi(rssi_raw) != BLE_STATUS_SUCCESS) {
458 return 0.0f;
459 }
460
461 // Some ST magic with rssi
462 uint8_t agc = rssi_raw[2] & 0xFF;
463 int rssi = (((int)rssi_raw[1] << 8) & 0xFF00) + (rssi_raw[0] & 0xFF);
464 if(rssi == 0 || agc > 11) {
465 val = -127.0;
466 } else {
467 val = agc * 6.0f - 127.0f;
468 while(rssi > 30) {
469 val += 6.0;
470 rssi >>= 1;
471 }
472 val += (float)((417 * rssi + 18080) >> 10);
473 }
474 return val;
475}
