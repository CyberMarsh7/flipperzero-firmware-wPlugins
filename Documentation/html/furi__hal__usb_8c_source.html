262static usbd_respond usb_descriptor_get(usbd_ctlreq* req, void** address, uint16_t* length) {
263 const uint8_t dtype = req->wValue >> 8;
264 const uint8_t dnumber = req->wValue & 0xFF;
265 const void* desc;
266 uint16_t len = 0;
267 if(usb.interface == NULL) return usbd_fail;
268
269 switch(dtype) {
270 case USB_DTYPE_DEVICE:
271 furi_thread_flags_set(furi_thread_get_id(usb.thread), UsbEventRequest);
272 if(usb.callback != NULL) {
273 usb.callback(FuriHalUsbStateEventDescriptorRequest, usb.callback_context);
274 }
275 desc = usb.interface->dev_descr;
276 break;
277 case USB_DTYPE_CONFIGURATION:
278 desc = usb.interface->cfg_descr;
279 len = ((struct usb_string_descriptor*)(usb.interface->cfg_descr))->wString[0];
280 break;
281 case USB_DTYPE_STRING:
282 if(dnumber == UsbDevLang) {
283 desc = &dev_lang_desc;
284 } else if((dnumber == UsbDevManuf) && (usb.interface->str_manuf_descr != NULL)) {
285 desc = usb.interface->str_manuf_descr;
286 } else if((dnumber == UsbDevProduct) && (usb.interface->str_prod_descr != NULL)) {
287 desc = usb.interface->str_prod_descr;
288 } else if((dnumber == UsbDevSerial) && (usb.interface->str_serial_descr != NULL)) {
289 desc = usb.interface->str_serial_descr;
290 } else
291 return usbd_fail;
292 break;
293 default:
294 return usbd_fail;
295 }
296 if(desc == NULL) return usbd_fail;
297
298 if(len == 0) {
299 len = ((struct usb_header_descriptor*)desc)->bLength;
300 }
301 *address = (void*)desc;
302 *length = len;
303 return usbd_ack;
304}

421static void usb_process_message(UsbApiEventMessage* message) {
422 switch(message->type) {
423 case UsbApiEventTypeSetConfig:
424 message->return_data->bool_value = usb_process_set_config(
425 message->data.interface.interface, message->data.interface.context);
426 break;
427 case UsbApiEventTypeGetConfig:
428 message->return_data->void_value = usb.interface;
429 break;
430 case UsbApiEventTypeGetConfigContext:
431 message->return_data->void_value = usb.interface_context;
432 break;
433 case UsbApiEventTypeLock:
434 FURI_LOG_I(TAG, "Mode lock");
435 usb.mode_lock = true;
436 break;
437 case UsbApiEventTypeUnlock:
438 FURI_LOG_I(TAG, "Mode unlock");
439 usb.mode_lock = false;
440 break;
441 case UsbApiEventTypeIsLocked:
442 message->return_data->bool_value = usb.mode_lock;
443 break;
444 case UsbApiEventTypeDisable:
445 usb_process_enable(false);
446 break;
447 case UsbApiEventTypeEnable:
448 usb_process_enable(true);
449 break;
450 case UsbApiEventTypeReinit:
451 usb_process_mode_reinit();
452 break;
453 case UsbApiEventTypeSetStateCallback:
454 usb.callback = message->data.state_callback.callback;
455 usb.callback_context = message->data.state_callback.context;
456 break;
457 }
458
459 api_lock_unlock(message->lock);

462static int32_t furi_hal_usb_thread(void* context) {
463 UNUSED(context);
464 uint8_t usb_wait_time = 0;
465
466 if(furi_message_queue_get_count(usb.queue) > 0) {
467 furi_thread_flags_set(furi_thread_get_id(usb.thread), UsbEventMessage);
468 }
469
470 while(true) {
471 uint32_t flags = furi_thread_flags_wait(USB_SRV_ALL_EVENTS, FuriFlagWaitAny, 500);
472
473 {
474 UsbApiEventMessage message;
475 if(furi_message_queue_get(usb.queue, &message, 0) == FuriStatusOk) {
476 usb_process_message(&message);
477 }
478 }
479
480 if((flags & FuriFlagError) == 0) {
481 if(flags & UsbEventReset) {
482 if(usb.enabled) {
483 usb.request_pending = true;
484 usb_wait_time = 0;
485 }
486 }
487 if(flags & UsbEventRequest) {
488 usb.request_pending = false;
489 }
490 } else if(usb.request_pending) {
491 usb_wait_time++;
492 if(usb_wait_time > 4) {
493 usb_process_mode_reinit();
494 usb.request_pending = false;
495 }
496 }
497 }
498 return 0;
